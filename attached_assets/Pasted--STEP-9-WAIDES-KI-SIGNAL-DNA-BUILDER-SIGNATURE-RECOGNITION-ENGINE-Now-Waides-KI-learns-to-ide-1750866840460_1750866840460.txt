✅ 1️⃣ waides_ki_dna_engine.py — DNA Generator for Signals

# waides_ki_dna_engine.py

import hashlib
import json

class WaidesKIDNAEngine:
    def __init__(self):
        self.registered_dna = {}

    def generate_dna(self, signal_data):
        # Use consistent fields to generate hash (acts as signal fingerprint)
        unique_string = f"{signal_data['trend']}|{signal_data['rsi']}|{signal_data['vwap_status']}|{signal_data['ema50']}"
        dna = hashlib.sha256(unique_string.encode()).hexdigest()[:16]
        return dna

    def register_dna(self, dna_id, strategy_info):
        self.registered_dna[dna_id] = strategy_info

    def get_dna_info(self, dna_id):
        return self.registered_dna.get(dna_id)

waides_ki_dna_engine = WaidesKIDNAEngine()

✅ 2️⃣ signal_signature_tracker.py — Clone & Mutation Detection

# signal_signature_tracker.py

import datetime

class SignalSignatureTracker:
    def __init__(self):
        self.seen_dna = {}  # DNA : [result_history]

    def record_result(self, dna, result):
        now = datetime.datetime.utcnow().isoformat()
        if dna not in self.seen_dna:
            self.seen_dna[dna] = []
        self.seen_dna[dna].append((result, now))

    def is_unstable(self, dna):
        results = self.seen_dna.get(dna, [])
        recent = results[-5:]
        losses = sum(1 for r in recent if r[0] == "loss")
        return len(recent) >= 3 and losses >= 3  # 3+ recent losses

    def get_history(self, dna):
        return self.seen_dna.get(dna, [])

signal_signature_tracker = SignalSignatureTracker()

✅ STEP 1: LIVE DATA FETCHERS (ETH Price, Volume, Trend)

async function fetchETHMarketData() {
  const res = await fetch("https://api.coingecko.com/api/v3/simple/price?ids=ethereum&vs_currencies=usd&include_market_cap=true&include_24hr_vol=true&include_24hr_change=true");
  const data = await res.json();
  return {
    price: data.ethereum.usd,
    volume: data.ethereum.usd_24h_vol,
    change: data.ethereum.usd_24h_change
  };
}

✅ STEP 3: DECISION-MAKING ENGINE (The “Brain”)

async function waidesKiEngine(wallet) {
  const market = await fetchETHMarketData();

  // Record current state
  waidesMemory.lastPrices.push(market.price);
  if (waidesMemory.lastPrices.length > 50) waidesMemory.lastPrices.shift();

  // Analyze trend
  const priceNow = market.price;
  const price5minAgo = waidesMemory.lastPrices[waidesMemory.lastPrices.length - 6];
  const trend = priceNow > price5minAgo ? 'UP' : priceNow < price5minAgo ? 'DOWN' : 'SIDE';

  // Inner vision + logic filter
  const strategy = decideTrade({
    trend,
    change: market.change,
    volume: market.volume,
    gainStreak: waidesMemory.gainStreak,
    failStreak: waidesMemory.failStreak,
    mode: wallet.activeBot === 'WaidbotPro' ? 'aggressive' : 'safe'
  });

  if (strategy.shouldTrade) {
    const result = await executeTrade(strategy.type, wallet);
    updateMemory(result);
  }
}

✅ STEP 4: STRATEGY FILTER (“decideTrade” Function)

function decideTrade({ trend, change, volume, gainStreak, failStreak, mode }) {
  if (mode === 'safe' && failStreak >= 2) return { shouldTrade: false };

  if (trend === 'UP' && change > 0.5) return { shouldTrade: true, type: 'BUY' };
  if (trend === 'DOWN' && change < -0.5 && mode === 'aggressive') return { shouldTrade: true, type: 'SELL' };
  if (trend === 'SIDE' && volume > 1000000000 && mode === 'aggressive') return { shouldTrade: true, type: 'SCALP' };

  return { shouldTrade: false };
}

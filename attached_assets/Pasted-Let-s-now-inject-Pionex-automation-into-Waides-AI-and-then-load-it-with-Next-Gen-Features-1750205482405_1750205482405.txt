### 🔌 STEP 3: Add `pionex_trade.py`

```python
# backend/pionex_trade.py

import time
import hmac
import hashlib
import base64
import requests
import os

API_KEY = os.getenv("PIONEX_API_KEY")
SECRET_KEY = os.getenv("PIONEX_SECRET_KEY")

BASE_URL = "https://api.pionex.com"

def generate_signature(timestamp, method, request_path, body=''):
    raw = f"{timestamp}{method}{request_path}{body}".encode()
    return base64.b64encode(
        hmac.new(SECRET_KEY.encode(), raw, hashlib.sha256).digest()
    ).decode()

def send_order(symbol, side, quantity, price):
    path = "/api/v1/order"
    timestamp = str(int(time.time() * 1000))

    body = {
        "symbol": symbol,
        "side": side.lower(),  # buy or sell
        "type": "limit",
        "quantity": quantity,
        "price": price
    }

    import json
    json_body = json.dumps(body)
    signature = generate_signature(timestamp, "POST", path, json_body)

    headers = {
        "PIONEX-KEY": API_KEY,
        "PIONEX-SIGN": signature,
        "PIONEX-TIMESTAMP": timestamp,
        "Content-Type": "application/json"
    }

    response = requests.post(BASE_URL + path, headers=headers, data=json_body)
    return response.json()
```

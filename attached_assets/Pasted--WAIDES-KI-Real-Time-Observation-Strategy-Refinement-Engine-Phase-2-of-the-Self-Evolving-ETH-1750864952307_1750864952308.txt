# waides_ki_observer.py

import random

class WaidesKIObserver:
    def __init__(self):
        self.eth_price = None
        self.ema50 = None
        self.ema200 = None
        self.vwap = None
        self.rsi = None

    def fetch_market_data(self):
        # Simulated data; you will later replace this with Binance/Bybit API
        self.eth_price = round(random.uniform(2480, 2525), 2)
        self.ema50 = round(random.uniform(2470, 2520), 2)
        self.ema200 = round(random.uniform(2500, 2550), 2)
        self.vwap = round(random.uniform(2480, 2520), 2)
        self.rsi = round(random.uniform(30, 70), 2)

    def analyze_structure(self):
        if self.ema50 > self.ema200:
            return "uptrend"
        elif self.ema50 < self.ema200:
            return "downtrend"
        return "range"

    def vwap_status(self):
        return "ABOVE" if self.eth_price > self.vwap else "BELOW"

    def get_indicators(self):
        return {
            "price": self.eth_price,
            "ema50": self.ema50,
            "ema200": self.ema200,
            "vwap": self.vwap,
            "rsi": self.rsi,
            "trend": self.analyze_structure(),
            "vwap_status": self.vwap_status()
        }

waides_ki_observer = WaidesKIObserver()

# waides_ki_signal_logger.py

import json
import os
import datetime

class SignalLogger:
    def __init__(self, file="waides_ki_signals.json"):
        self.file = file
        self.data = self.load()

    def load(self):
        if os.path.exists(self.file):
            with open(self.file, "r") as f:
                return json.load(f)
        return []

    def log_signal(self, signal_strength, indicators, strategy_id):
        entry = {
            "timestamp": datetime.datetime.utcnow().isoformat(),
            "strategy_id": strategy_id,
            "signal_strength": signal_strength,
            "indicators": indicators
        }
        self.data.append(entry)
        with open(self.file, "w") as f:
            json.dump(self.data, f, indent=2)

signal_logger = SignalLogger()

from waides_ki_memory import waides_ki_memory
from waides_ki_strategy import WaidesKIStrategy
from waides_ki_observer import waides_ki_observer
from waides_ki_signal_logger import signal_logger
import datetime
import time
import random

class WaidesKI:
    def __init__(self):
        self.capital = 1000
        self.strategy_engine = WaidesKIStrategy(waides_ki_memory)
        self.trade_log = []

    def assess_signal_strength(self, indicators):
        score = 0
        if indicators["trend"] == "uptrend" and indicators["vwap_status"] == "ABOVE" and indicators["rsi"] > 50:
            score += 3
        elif indicators["trend"] == "downtrend" and indicators["vwap_status"] == "BELOW" and indicators["rsi"] < 50:
            score += 3
        elif indicators["trend"] == "range":
            score += 1
        if abs(indicators["rsi"] - 50) > 10:
            score += 1
        return score

    def decide_and_trade(self):
        waides_ki_observer.fetch_market_data()
        indicators = waides_ki_observer.get_indicators()

        strategy_id = self.strategy_engine.get_strategy_id(
            indicators["trend"],
            indicators["rsi"],
            indicators["vwap_status"]
        )

        signal_strength = self.assess_signal_strength(indicators)
        signal_logger.log_signal(signal_strength, indicators, strategy_id)

        if signal_strength >= 3:
            trade = self.execute_trade("buy" if indicators["trend"] == "uptrend" else "sell", indicators["price"], strategy_id)
            self.evaluate_result(trade)
        else:
            print("⚠️ Weak signal ignored. Strength:", signal_strength)

    def execute_trade(self, direction, entry_price, strategy_id):
        stop_loss = entry_price - 15 if direction == "buy" else entry_price + 15
        take_profit = entry_price + 30 if direction == "buy" else entry_price - 30
        trade = {
            "time": datetime.datetime.utcnow().isoformat(),
            "direction": direction,
            "entry_price": entry_price,
            "stop_loss": stop_loss,
            "take_profit": take_profit,
            "strategy_id": strategy_id
        }
        self.trade_log.append(trade)
        print(f"✅ TRADE: {direction.upper()} at ${entry_price} | Strategy: {strategy_id}")
        return trade

    def evaluate_result(self, trade):
        current_price = random.uniform(2475, 2550)
        self.strategy_engine.evaluate_trade_result(trade, current_price)

    def run(self, loops=5):
        for _ in range(loops):
            self.decide_and_trade()
            print("---")
            time.sleep(1)

if __name__ == "__main__":
    bot = WaidesKI()
    bot.run(10)

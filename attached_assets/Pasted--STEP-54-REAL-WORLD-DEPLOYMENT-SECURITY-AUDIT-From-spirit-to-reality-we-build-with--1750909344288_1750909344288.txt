3️⃣ Risk Controls & Compliance — risk_manager.py

class RiskManager:
    def __init__(self, max_daily_loss=0.05, max_trade_size=0.02):
        self.daily_loss_limit = max_daily_loss
        self.max_trade_size = max_trade_size
        self.daily_loss = 0.0

    def can_trade(self, account_balance, trade_amount):
        if trade_amount > account_balance * self.max_trade_size:
            return False
        if self.daily_loss >= account_balance * self.daily_loss_limit:
            return False
        return True

    def record_result(self, pnl, account_balance):
        if pnl < 0:
            self.daily_loss += abs(pnl)

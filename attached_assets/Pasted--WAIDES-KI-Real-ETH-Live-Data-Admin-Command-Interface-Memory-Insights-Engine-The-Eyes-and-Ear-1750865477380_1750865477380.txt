# waides_ki_live_feed.py

import requests

class WaidesKILiveFeed:
    def __init__(self):
        self.endpoint = "https://api.binance.com/api/v3/klines"

    def fetch_eth_data(self):
        try:
            params = {
                "symbol": "ETHUSDT",
                "interval": "1m",
                "limit": 50
            }
            res = requests.get(self.endpoint, params=params).json()
            closes = [float(x[4]) for x in res]
            highs = [float(x[2]) for x in res]
            lows = [float(x[3]) for x in res]

            current_price = closes[-1]
            ema50 = sum(closes[-50:]) / 50
            ema200 = sum(closes + closes[:150]) / 200  # Mock EMA200
            vwap = sum((h + l + c) / 3 for h, l, c in zip(highs, lows, closes)) / len(closes)

            rsi = self.calculate_rsi(closes)

            return {
                "price": round(current_price, 2),
                "ema50": round(ema50, 2),
                "ema200": round(ema200, 2),
                "vwap": round(vwap, 2),
                "rsi": round(rsi, 2)
            }

        except Exception as e:
            print("Live feed error:", e)
            return None

    def calculate_rsi(self, closes, period=14):
        gains, losses = [], []
        for i in range(1, period + 1):
            diff = closes[-i] - closes[-i - 1]
            gains.append(max(0, diff))
            losses.append(max(0, -diff))
        avg_gain = sum(gains) / period
        avg_loss = sum(losses) / period
        rs = avg_gain / avg_loss if avg_loss != 0 else 1
        return 100 - (100 / (1 + rs))

waides_ki_live_feed = WaidesKILiveFeed()

# waides_ki_admin.py

import json
from waides_ki_memory import waides_ki_memory

class WaidesKIAdmin:
    def __init__(self, memory):
        self.memory = memory

    def command(self, cmd):
        if cmd == "/status":
            return self.get_status()
        elif cmd == "/memory":
            return self.get_memory()
        elif cmd == "/strategy":
            return self.get_strategies()
        else:
            return "❓ Unknown command"

    def get_status(self):
        hist = self.memory.memory["history"]
        wins = sum(1 for h in hist if h["result"] == "win")
        losses = sum(1 for h in hist if h["result"] == "loss")
        return f"📊 STATUS: Wins: {wins}, Losses: {losses}, Total: {len(hist)}"

    def get_memory(self):
        return json.dumps(self.memory.memory["mistakes"], indent=2)

    def get_strategies(self):
        strategies = self.memory.memory["strategies"]
        ranked = sorted(strategies.items(), key=lambda x: x[1]["win"], reverse=True)
        top = ranked[:3]
        return "\n".join([f"{sid} → {data['win']}W / {data['loss']}L" for sid, data in top])

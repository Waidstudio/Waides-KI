✅ 1️⃣ waides_api_gateway.py — External Access Brain

# waides_api_gateway.py

from flask import Blueprint, request, jsonify
from waides_genome_engine import waides_genome_engine
from waides_ki_dna_engine import waides_ki_dna_engine
from waides_ki_self_repair import waides_ki_self_repair
from waides_root_memory import waides_root_memory
from signal_signature_tracker import signal_signature_tracker

waides_api_gateway = Blueprint("waides_api_gateway", __name__)

# Trusted API keys (later move this to a DB or secure file)
TRUSTED_KEYS = {
    "demo123": "Afriwaid Studio",
    "alpha998": "Konsmia Market App"
}

@waides_api_gateway.route("/api/strategy", methods=["POST"])
def get_strategy():
    api_key = request.headers.get("X-API-KEY")
    if api_key not in TRUSTED_KEYS:
        return jsonify({"error": "Unauthorized"}), 403

    payload = request.json
    indicators = payload.get("indicators")
    base_id = TRUSTED_KEYS[api_key] + "-" + str(indicators["price"])

    # Generate a strategy using mutation engine
    new_strategy_id = waides_genome_engine.create_new_strategy(base_id, indicators)
    new_dna = waides_genome_engine.generated_strategies[new_strategy_id]["dna"]

    # Check risk
    if signal_signature_tracker.is_unstable(new_dna):
        return jsonify({"status": "blocked", "reason": "DNA unstable"})

    # Register memory and return strategy
    waides_root_memory.register_strategy(new_strategy_id, new_dna, "untested")
    return jsonify({
        "strategy_id": new_strategy_id,
        "dna": new_dna,
        "code": waides_genome_engine.generated_strategies[new_strategy_id]["strategy_code"]
    })

✅ 3️⃣ Sample API Client (Python Script)

You can now let partners trade ETH using Waides KI like this:

import requests

url = "https://waideski.com/api/strategy"
headers = {
    "X-API-KEY": "demo123",
    "Content-Type": "application/json"
}
data = {
    "indicators": {
        "trend": "uptrend",
        "rsi": 66,
        "vwap_status": "ABOVE",
        "price": 2533,
        "ema50": 2505
    }
}
response = requests.post(url, json=data, headers=headers)
print(response.json())

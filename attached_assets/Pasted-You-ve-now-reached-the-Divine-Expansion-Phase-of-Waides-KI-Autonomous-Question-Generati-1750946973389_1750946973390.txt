## ✅ STEP 1: Build `QuestionSeeder.js` (Smart Generator)

```js
const categories = [
  "ethereum",
  "trading",
  "money",
  "emotions",
  "spirit",
  "ai",
  "konsmia",
  "waides ki",
  "fear",
  "purpose",
  "time",
  "dream",
  "smai",
  "bots",
  "data",
  "privacy"
];

const questionTemplates = [
  "what is the future of ___?",
  "how does ___ affect trading?",
  "can ___ be predicted?",
  "why is ___ important?",
  "is ___ safe?",
  "how to learn ___?",
  "examples of ___ in action",
  "does ___ work during ETH crashes?",
  "can bots use ___?",
  "how does Konsmia treat ___?",
  "what does Waides KI think of ___?"
];

export function generateQuestions(n = 100) {
  const questions = [];

  for (let i = 0; i < n; i++) {
    const cat = categories[Math.floor(Math.random() * categories.length)];
    const temp = questionTemplates[Math.floor(Math.random() * questionTemplates.length)];
    const q = temp.replace("___", cat);
    questions.push(q);
  }

  return questions;
}
```

## ✅ STEP 2: Connect to `KnowledgeLoader.js`

Here we store the generated questions, and (optionally) add blank answers.

```js
import UKC from "./UKC";

export function saveQuestionsToUKC(questions) {
  for (let q of questions) {
    const lower = q.toLowerCase();
    const topic = extractCategory(lower);
    if (!UKC.categories[topic]) UKC.categories[topic] = {};
    if (!UKC.categories[topic][lower]) {
      UKC.categories[topic][lower] = "🌱 This is a seed question. Answer coming soon.";
    }
  }
}

// Simple extractor
function extractCategory(q) {
  for (const cat of Object.keys(UKC.categories)) {
    if (q.includes(cat)) return cat;
  }
  return "misc";
}
```

1. Metrics Collection (metrics_collector.py)

from prometheus_client import Gauge, start_http_server
import psutil
import time

cpu_gauge = Gauge('waides_cpu_percent', 'CPU usage')
mem_gauge = Gauge('waides_mem_percent', 'Memory usage')
loop_latency = Gauge('waides_loop_latency_ms', 'Event loop latency')

def collect_metrics():
    start_http_server(8001)
    while True:
        t0 = time.time()
        cpu_gauge.set(psutil.cpu_percent())
        mem_gauge.set(psutil.virtual_memory().percent)
        # simulate measuring event loop
        t1 = time.time()
        loop_latency.set((t1 - t0)*1000)
        time.sleep(5)

if __name__ == "__main__":
    collect_metrics()

B) On‑Chain DAO Integration

1. Solidity DAO Smart Contract (WaidesDAO.sol)

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract WaidesDAO {
    struct Proposal {
        address proposer;
        string description;
        uint256 voteCountYes;
        uint256 voteCountNo;
        mapping(address => bool) voted;
        bool executed;
    }

    mapping(uint256 => Proposal) public proposals;
    uint256 public proposalCount;
    mapping(address => uint256) public stake;

    modifier onlyStaker() {
        require(stake[msg.sender] > 0, "Stake required");
        _;
    }

    function stakeTokens(uint256 amount) external payable {
        stake[msg.sender] += amount;
    }

    function propose(string calldata description) external onlyStaker returns (uint256) {
        proposalCount++;
        Proposal storage p = proposals[proposalCount];
        p.proposer = msg.sender;
        p.description = description;
        return proposalCount;
    }

    function vote(uint256 id, bool support) external onlyStaker {
        Proposal storage p = proposals[id];
        require(!p.voted[msg.sender], "Already voted");
        p.voted[msg.sender] = true;
        if (support) p.voteCountYes++;
        else p.voteCountNo++;
    }

    function execute(uint256 id) external {
        Proposal storage p = proposals[id];
        require(!p.executed, "Already executed");
        require(p.voteCountYes > p.voteCountNo, "Insufficient support");
        p.executed = true;
    }
}

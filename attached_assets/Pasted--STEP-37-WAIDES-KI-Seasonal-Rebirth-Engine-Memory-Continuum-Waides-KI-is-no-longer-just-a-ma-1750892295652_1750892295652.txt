✅ 1️⃣ seasonal_engine.py

import json, os
from datetime import datetime
from memory_vault import memory_vault
from waides_shadow_lab import waides_shadow_lab
from waides_strategy_vault import waides_strategy_vault

class SeasonalRebirthEngine:
    def __init__(self, state_file="current_season.json"):
        self.state_file = state_file
        self.load_state()

    def load_state(self):
        if not os.path.exists(self.state_file):
            self.state = {
                "season_index": 0,
                "start_date": datetime.utcnow().isoformat()
            }
            self.save_state()
        else:
            with open(self.state_file) as f:
                self.state = json.load(f)

    def save_state(self):
        with open(self.state_file, 'w') as f:
            json.dump(self.state, f, indent=4)

    def check_cycle(self):
        last = datetime.fromisoformat(self.state["start_date"])
        now = datetime.utcnow()
        return (now - last).days >= 90

    def begin_new_season(self):
        season_id = self.state["season_index"]
        name = memory_vault.get_season_name(season_id)
        dna_snap = waides_strategy_vault.all()

        # Save memory of current season
        memory_vault.archive_season(name, dna_snap)

        # Reset strategy vault and shadow lab
        waides_strategy_vault.vault = {}
        waides_shadow_lab.shadow_strategies = {}

        # Evolve to next season
        self.state["season_index"] += 1
        self.state["start_date"] = datetime.utcnow().isoformat()
        self.save_state()

        return {"season": name, "status": "Rebirth Completed"}

seasonal_rebirth_engine = SeasonalRebirthEngine()

✅ 2️⃣ memory_vault.py

import json, os
from datetime import datetime
import random

class MemoryVault:
    def __init__(self, log_file="waides_memories.json", name_file="season_names.json"):
        self.log_file = log_file
        self.name_file = name_file

        if not os.path.exists(self.log_file):
            with open(self.log_file, 'w') as f:
                json.dump([], f)

        if not os.path.exists(self.name_file):
            with open(self.name_file, 'w') as f:
                json.dump(["Sha’Tuun", "Mir’Vel", "Tal’Anar", "Al’Zaan", "Dru’Kai"], f)

    def get_season_name(self, index):
        with open(self.name_file) as f:
            names = json.load(f)
        return names[index % len(names)]

    def archive_season(self, name, dna_snapshot):
        with open(self.log_file, 'r') as f:
            history = json.load(f)

        history.append({
            "season": name,
            "timestamp": datetime.utcnow().isoformat(),
            "strategies": dna_snapshot
        })

        with open(self.log_file, 'w') as f:
            json.dump(history, f, indent=4)

memory_vault = MemoryVault()

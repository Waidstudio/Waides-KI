✅ 1️⃣ entangled_presence_mesh.py — Shared Presence Exchange

import requests
from presence_orchestrator import presence_orchestrator

class PresenceMesh:
    def __init__(self, peers):
        self.peers = peers
        self.mesh_states = {}

    def broadcast(self):
        data = presence_orchestrator.evaluate()
        for url in self.peers:
            try:
                requests.post(f"{url}/api/mesh/state", json=data, timeout=1)
            except:
                pass

    def gather(self):
        self.mesh_states = {}
        for url in self.peers:
            try:
                resp = requests.get(f"{url}/api/mesh/state", timeout=1).json()
                self.mesh_states[url] = resp
            except:
                continue
        return self.mesh_states

presence_mesh = PresenceMesh(peers=["https://node1", "https://node2"])

✅ 2️⃣ empathy_weight_manager.py — Weighted Node Trust

class EmpathyWeightManager:
    def __init__(self):
        self.node_scores = {}  # node_url: win_rate

    def update_score(self, node_url, score):
        self.node_scores[node_url] = score

    def weight(self, node_url):
        return self.node_scores.get(node_url, 0.5)  # default 0.5

emp_manager = EmpathyWeightManager()

✅ 3️⃣ collective_trade_conductor.py — Consensus Vote Engine

from entangled_presence_mesh import presence_mesh
from empathy_weight_manager import emp_manager
from presence_orchestrator import presence_orchestrator

class CollectiveConductor:
    def propose_trade(self):
        local = presence_orchestrator.evaluate()
        mesh = presence_mesh.gather()
        total_weight = 1.0  # self
        votes = []
        
        local_vote = local["order_pressure"]
        votes.append((local_vote, 1.0))

        for node, state in mesh.items():
            w = emp_manager.weight(node)
            votes.append((state["order_pressure"], w))
            total_weight += w

        # Tally weighted votes
        tally = {}
        for vote, w in votes:
            tally[vote] = tally.get(vote, 0) + w

        # Decide consensus
        best = max(tally, key=lambda k: tally[k])
        confidence = tally[best] / total_weight
        return {"decision": best, "confidence": confidence, "votes": tally}

collective_conductor = CollectiveConductor()

✅ 4️⃣ humanization_interpreter.py — Community Trade Language

class HumanizationInterpreter:
    def interpret(self, proposal):
        d = proposal["decision"]
        c = proposal["confidence"]
        if d == "buy_pressure" and c > 0.7:
            return f"✅ Community says BUY (confidence {c*100:.0f}%)"
        if d == "sell_pressure" and c > 0.7:
            return f"🚫 Community leans SELL (confidence {c*100:.0f}%)"
        return f"🤔 Not aligned enough ({d} at {c*100:.0f}%)"

humanizer = HumanizationInterpreter()

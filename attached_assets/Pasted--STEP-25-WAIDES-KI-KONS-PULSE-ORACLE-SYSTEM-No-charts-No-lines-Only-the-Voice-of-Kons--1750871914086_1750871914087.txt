# kons_pulse_oracle.py

from frequency_interpreter import frequency_interpreter
from core.utils.price_feed import fetch_eth_candles
import random

class KonsPulseOracle:
    def __init__(self):
        self.last_forecast = None

    def analyze(self):
        candles_4h = fetch_eth_candles("4h")
        candles_1d = fetch_eth_candles("1d")
        candles_1w = fetch_eth_candles("1w")

        def trend(c): return "up" if c[-1]["close"] > c[0]["close"] else "down"

        forecast = {
            "4h": trend(candles_4h),
            "1d": trend(candles_1d),
            "1w": trend(candles_1w)
        }

        kons_messages = {
            tf: frequency_interpreter.interpret(dir, tf)
            for tf, dir in forecast.items()
        }

        self.last_forecast = kons_messages
        return kons_messages

kons_pulse_oracle = KonsPulseOracle()

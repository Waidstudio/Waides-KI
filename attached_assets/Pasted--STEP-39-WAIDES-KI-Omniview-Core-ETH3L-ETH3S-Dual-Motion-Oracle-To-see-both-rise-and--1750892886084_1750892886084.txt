✅ 1️⃣ price_feed_binance.py — Binance ETH Data Feed

import requests

def get_binance_data(symbol="ETHUSDT", interval="15m", limit=100):
    url = f"https://api.binance.com/api/v3/klines?symbol={symbol}&interval={interval}&limit={limit}"
    res = requests.get(url).json()
    candles = [{
        "open_time": c[0],
        "open": float(c[1]),
        "high": float(c[2]),
        "low": float(c[3]),
        "close": float(c[4]),
        "volume": float(c[5])
    } for c in res]
    return candles

✅ 2️⃣ eth_trend_profiler.py — Detect ETH Up or Down

def detect_trend(candles):
    closes = [c["close"] for c in candles[-10:]]
    avg_change = (closes[-1] - closes[0]) / closes[0]
    
    if abs(avg_change) < 0.005:
        return "sideways"
    return "up" if avg_change > 0 else "down"

✅ 3️⃣ omniview_oracle.py — Waides’ 3rd Eye

from price_feed_binance import get_binance_data
from eth_trend_profiler import detect_trend

class OmniviewOracle:
    def __init__(self):
        self.required_agreement = 2  # How many timeframes must agree

    def scan_all(self):
        intervals = ["15m", "1h", "4h", "1d"]
        results = []

        for i in intervals:
            candles = get_binance_data("ETHUSDT", interval=i)
            trend = detect_trend(candles)
            results.append(trend)

        up_votes = results.count("up")
        down_votes = results.count("down")

        if up_votes >= self.required_agreement:
            return "ETH3L"
        elif down_votes >= self.required_agreement:
            return "ETH3S"
        return "no_trade"

✅ 1️⃣ waides_shadow_simulator.py — Core Shadow Engine

# waides_shadow_simulator.py

import datetime
from waides_ki_dna_engine import waides_ki_dna_engine
from signal_signature_tracker import signal_signature_tracker

class WaidesShadowSimulator:
    def __init__(self):
        self.shadow_log = []

    def simulate_alternates(self, current_indicators, real_dna):
        variants = []

        # Try 3 shadow setups with altered indicators
        for i in range(3):
            alt_rsi = current_indicators["rsi"] + (-5 + i * 5)  # e.g. -5, 0, +5
            alt_data = {
                **current_indicators,
                "rsi": alt_rsi
            }

            alt_dna = waides_ki_dna_engine.generate_dna(alt_data)
            if alt_dna == real_dna:
                continue

            # Fake a result for now (replace with real test simulator)
            fake_profit = 5 - abs(alt_rsi - current_indicators["rsi"])  # closer to real = better
            result_status = "win" if fake_profit > 0 else "loss"

            signal_signature_tracker.record_result(alt_dna, result_status)
            self.shadow_log.append({
                "time": datetime.datetime.utcnow().isoformat(),
                "variant_dna": alt_dna,
                "rsi": alt_rsi,
                "result": result_status,
                "simulated_profit": fake_profit
            })

    def get_shadow_log(self):
        return self.shadow_log

waides_shadow_simulator = WaidesShadowSimulator()

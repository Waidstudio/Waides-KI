✅ 1️⃣ konsigil_engine.py — Konsigil Generator

# konsigil_engine.py

import uuid
import hashlib

class KonsigilEngine:
    def generate(self, trade_context):
        raw_string = f"{trade_context['pattern']}-{trade_context['emotion']}-{uuid.uuid4()}"
        glyph = hashlib.sha256(raw_string.encode()).hexdigest()[:12]
        return {
            "konsigil": glyph,
            "origin_pattern": trade_context["pattern"],
            "emotion_shade": trade_context["emotion"],
            "timestamp": trade_context.get("time")
        }

konsigil_engine = KonsigilEngine()

✅ 2️⃣ glyph_memory_tree.py — Store Trade Glyphs

# glyph_memory_tree.py

class GlyphMemoryTree:
    def __init__(self):
        self.tree = []  # [{konsigil, result, power_score, notes}]

    def stamp(self, glyph_data, result, notes=""):
        glyph_data["result"] = result
        glyph_data["power_score"] = self.analyze(result)
        glyph_data["notes"] = notes
        self.tree.append(glyph_data)

    def analyze(self, result):
        if result > 0:
            return 1.0
        elif result == 0:
            return 0.5
        else:
            return -1.0

    def find(self, konsigil):
        return next((g for g in self.tree if g["konsigil"] == konsigil), None)

glyph_memory_tree = GlyphMemoryTree()

✅ 3️⃣ sigil_oracle.py — Read + Learn From Glyphs

# sigil_oracle.py

from glyph_memory_tree import glyph_memory_tree

class SigilOracle:
    def scan(self, pattern, emotion):
        known = [g for g in glyph_memory_tree.tree if g["origin_pattern"] == pattern and g["emotion_shade"] == emotion]
        if not known:
            return {"confidence": 0.5, "message": "No prior glyph match"}
        
        avg = sum(g["power_score"] for g in known) / len(known)
        return {
            "confidence": avg,
            "glyphs_checked": len(known),
            "message": "Scan complete"
        }

sigil_oracle = SigilOracle()

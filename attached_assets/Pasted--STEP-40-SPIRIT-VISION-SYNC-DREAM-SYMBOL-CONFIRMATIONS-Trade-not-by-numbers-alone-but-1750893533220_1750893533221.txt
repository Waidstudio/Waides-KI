import random
from konslang_dictionary import KONSLANG_SYMBOLS

class SpiritOracle:
    def __init__(self):
        self.last_symbol = None

    def generate_symbol(self, trend):
        if trend == "up":
            symbol = random.choice(["SHAI'LOR", "KAYL'RUH"])
        elif trend == "down":
            symbol = random.choice(["MYRAL'ZEN", "ULTH'RIK"])
        else:
            symbol = "SEN'TUAR"

        self.last_symbol = symbol
        return {"symbol": symbol, "meaning": KONSLANG_SYMBOLS[symbol]}

import requests

def get_eth_price():
    url = "https://api.coingecko.com/api/v3/simple/price?ids=ethereum&vs_currencies=usd"
    return requests.get(url).json()['ethereum']['usd']

### 🧠 `backend/kons_powa_comm.py`

```python
import datetime, random, hashlib
from .eth_connector import get_eth_price

KONS_TITLES = ["ElaitReign","SmaiBond","TimeKeeper"]

def generate_kons_signal(price):
    ts = datetime.datetime.utcnow().strftime("%H%M")
    raw = f"{price}-{ts}-{random.choice(KONS_TITLES)}"
    return hashlib.sha512(raw.encode()).hexdigest()[:24]

def divine_eth_intent():
    p = get_eth_price()
    sig = generate_kons_signal(p)
    val = sum(ord(c) for c in sig[-4:]) % 100
    if val>=75:
        return {"action":"BUY_LONG","timeframe":"5:30–9:00 PM WAT","reason":"Momentum Uptrend","signal":sig}
    elif val>=40:
        return {"action":"SELL_SHORT","timeframe":"1:00–4:00 AM WAT","reason":"Short Pullback","signal":sig}
    else:
        return {"action":"NO_TRADE","timeframe":"Observe","reason":"Uncertain","signal":sig}
```

### 🔁 `backend/pionex_bot.py`

```python
from pionex_python.restful.Orders import Orders
import os
from .kons_powa_comm import divine_eth_intent

key, secret = os.getenv("PIONEX_API_KEY"), os.getenv("PIONEX_SECRET_KEY")
orders = Orders(key, secret)

def run_bot():
    res = divine_eth_intent()
    if res["action"]=="BUY_LONG":
        orders.new_order({"symbol":"ETH_USDT","side":"BUY","type":"MARKET","amount":"0.01"})
    elif res["action"]=="SELL_SHORT":
        orders.new_order({"symbol":"ETH_USDT","side":"SELL","type":"MARKET","amount":"0.01"})
    return res
```

📌 **Note:** You'll need to `pip install pionex_python` ([github.com][1])

### ⚙️ `backend/app.py`

```python
from flask import Flask, jsonify
from .kons_powa_comm import divine_eth_intent
from .pionex_bot import run_bot

app = Flask(__name__)

@app.route('/api/signal', methods=['GET'])
def signal():
    return jsonify(divine_eth_intent())

@app.route('/api/auto-trade', methods=['POST'])
def auto_trade():
    res = run_bot()
    return jsonify({"executed": res})
```

## 🖥️ Step 2: Frontend Display (`frontend/App.jsx`)

```jsx
import React, {useEffect, useState} from 'react';
export default function App(){
  const [msg, set)=>useState(null);
  useEffect(()=>{
    fetch('/api/signal').then(r=>r.json()).then(set);
  },[]);
  return (
    <div className="main">
      <h1>🌌 Waides AI Kons Powa Bot</h1>
      {msg && (
        <div className="card">
          <h2>Action: {msg.action}</h2>
          <p>Clock: {msg.timeframe}</p>
          <p>Reason: {msg.reason}</p>
          <p>🔑 Signal: {msg.signal}</p>
          <button onClick={()=>fetch('/api/auto-trade',{method:'POST'}).then(r=>r.json()).then(a=>alert(a.executed.action+' executed'))}>Execute Trade</button>
        </div>
      )}
    </div>
  );
}
```

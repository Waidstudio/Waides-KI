✅ 3️⃣ vision_alignment_index.py — Are We All Seeing the Same Thing?

from collections import defaultdict
from datetime import datetime, timedelta

class VisionAlignmentIndex:
    def __init__(self):
        self.symbols = defaultdict(list)

    def add(self, signal):
        now = datetime.utcnow()
        self.symbols[signal["symbol"]].append(now)

    def top_symbol(self):
        threshold_time = datetime.utcnow() - timedelta(minutes=30)
        count_map = {
            sym: [t for t in times if t >= threshold_time]
            for sym, times in self.symbols.items()
        }
        return max(count_map.items(), key=lambda x: len(x[1]), default=(None, []))

vision_alignment_index = VisionAlignmentIndex()

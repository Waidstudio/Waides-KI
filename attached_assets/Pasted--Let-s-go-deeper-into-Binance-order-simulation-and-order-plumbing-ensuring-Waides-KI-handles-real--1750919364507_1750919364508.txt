1️⃣ Order Simulator — order_simulator.py

This module emulates Binance trades in a safe, test environment for dry runs and strategy validation.

import time
from datetime import datetime

class OrderSimulator:
    def __init__(self, starting_balance=10000):
        self.balance = starting_balance
        self.position = {}
        self.history = []

    def buy(self, symbol, quote_amount):
        price = self._get_price(symbol)
        qty = quote_amount / price
        self.balance -= quote_amount
        self.position[symbol] = self.position.get(symbol, 0) + qty
        trade = self._record("BUY", symbol, qty, price)
        return trade

    def sell(self, symbol, qty=None):
        held_qty = self.position.get(symbol, 0)
        qty = qty or held_qty
        if qty > held_qty:
            qty = held_qty
        price = self._get_price(symbol)
        amount = qty * price
        self.balance += amount
        self.position[symbol] = held_qty - qty
        trade = self._record("SELL", symbol, qty, price)
        return trade

    def _get_price(self, symbol):
        # Ideal: fetch from Binance testnet or use dummy data
        return 2500.0  # placeholder

    def _record(self, side, symbol, qty, price):
        trade = {
            "id": len(self.history) + 1,
            "side": side,
            "symbol": symbol,
            "qty": qty,
            "price": price,
            "time": datetime.utcnow().isoformat(),
            "balance": self.balance
        }
        self.history.append(trade)
        return trade

2️⃣ Order Plumbing Layer — order_manager.py

Interfaces between OrderSimulator and real Binance client, allowing plugs for either.

class OrderManager:
    def __init__(self, real_client=None, simulate=False):
        self.simulate = simulate
        self.client = real_client
        from order_simulator import order_simulator
        self.sim = order_simulator

    def buy(self, symbol, amount):
        if self.simulate:
            return self.sim.buy(symbol, quote_amount=amount)
        else:
            return self.client.buy(symbol, amount)

    def sell(self, symbol, qty):
        if self.simulate:
            return self.sim.sell(symbol, qty=qty)
        else:
            return self.client.sell(symbol, qty=qty)

order_manager = OrderManager(simulate=True)  # default to simulate mode

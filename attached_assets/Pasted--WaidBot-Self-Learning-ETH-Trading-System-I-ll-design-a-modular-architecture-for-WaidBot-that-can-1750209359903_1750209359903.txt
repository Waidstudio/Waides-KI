# WaidBot: Self-Learning ETH Trading System

I'll design a modular architecture for WaidBot that can integrate multiple exchange APIs (starting with Pionex) for training, with a machine learning core that evolves its trading strategies.

## Complete Implementation (Python)

```python
import ccxt
import numpy as np
import pandas as pd
from sklearn.ensemble import RandomForestRegressor
from sklearn.model_selection import train_test_split
from datetime import datetime
import json
import time
import os
from dotenv import load_dotenv

# Load environment variables
load_dotenv()

class WaidBot:
    def __init__(self):
        self.exchanges = {
            'pionex': self._init_pionex(),
            # Can add more exchanges later
        }
        self.models = {
            'trend': None,
            'volatility': None,
            'reversal': None
        }
        self.history = pd.DataFrame()
        self.balance = 1000  # Starting balance in USDT
        self.eth_held = 0
        self.trade_history = []
        
    def _init_pionex(self):
        return ccxt.pionex({
            'apiKey': os.getenv('PIONEX_API_KEY'),
            'secret': os.getenv('PIONEX_API_SECRET'),
            'enableRateLimit': True
        })
    
    def fetch_market_data(self, exchange_name='pionex', symbol='ETH/USDT', timeframe='1h', limit=100):
        """Fetch historical data from exchange"""
        exchange = self.exchanges[exchange_name]
        ohlcv = exchange.fetch_ohlcv(symbol, timeframe, limit=limit)
        df = pd.DataFrame(ohlcv, columns=['timestamp', 'open', 'high', 'low', 'close', 'volume'])
        df['timestamp'] = pd.to_datetime(df['timestamp'], unit='ms')
        return df
    
    def preprocess_data(self, df):
        """Create features for machine learning"""
        df['returns'] = df['close'].pct_change()
        df['volatility'] = df['returns'].rolling(5).std()
        df['trend'] = df['close'].rolling(10).mean()
        df['rsi'] = self._calculate_rsi(df['close'], 14)
        df = df.dropna()
        return df
    
    def _calculate_rsi(self, series, period):
        delta = series.diff()
        gain = delta.where(delta > 0, 0)
        loss = -delta.where(delta < 0, 0)
        
        avg_gain = gain.rolling(period).mean()
        avg_loss = loss.rolling(period).mean()
        
        rs = avg_gain / avg_loss
        return 100 - (100 / (1 + rs))
    
    def train_models(self):
        """Train ML models on historical data"""
        if self.history.empty:
            self.history = self.fetch_market_data()
        
        df = self.preprocess_data(self.history)
        
        # Features and targets
        X = df[['volatility', 'trend', 'rsi']]
        y_trend = (df['close'].shift(-1) > df['close']).astype(int)  # 1 if price goes up
        y_volatility = df['volatility'].shift(-1)
        
        # Train trend prediction model
        X_train, X_test, y_train, y_test = train_test_split(X, y_trend, test_size=0.2)
        self.models['trend'] = RandomForestRegressor()
        self.models['trend'].fit(X_train, y_train)
        
        # Train volatility prediction model
        X_train, X_test, y_train, y_test = train_test_split(X, y_volatility, test_size=0.2)
        self.models['volatility'] = RandomForestRegressor()
        self.models['volatility'].fit(X_train, y_train)
        
        print("Models trained successfully!")
    
    def make_decision(self):
        """Make trading decision based on current market conditions"""
        latest_data = self.fetch_market_data(limit=1)
        processed = self.preprocess_data(latest_data)
        
        if processed.empty:
            return "no_action"
        
        features = processed[['volatility', 'trend', 'rsi']].iloc[-1].values.reshape(1, -1)
        
        trend_prob = self.models['trend'].predict(features)[0]
        volatility = self.models['volatility'].predict(features)[0]
        
        current_price = processed['close'].iloc[-1]
        usdt_per_trade = self.balance * 0.1  # Risk 10% per trade
        
        if trend_prob > 0.6 and volatility < 0.05:
            # Buy signal - strong trend, low volatility
            amount = usdt_per_trade / current_price
            return ("buy", amount)
        elif trend_prob < 0.4 or volatility > 0.1:
            # Sell signal - weak trend or high volatility
            if self.eth_held > 0:
                return ("sell", self.eth_held * 0.5)  # Sell half position
        return "hold"
    
    def execute_trade(self, exchange_name='pionex', symbol='ETH/USDT'):
        """Execute the trading decision"""
        decision, amount = self.make_decision()
        exchange = self.exchanges[exchange_name]
        
        try:
            if decision == "buy":
                order = exchange.create_market_buy_order(symbol, amount)
                self.balance -= amount * order['price']
                self.eth_held += amount
                self.trade_history.append(order)
                print(f"Bought {amount} ETH at {order['price']}")
            elif decision == "sell":
                order = exchange.create_market_sell_order(symbol, amount)
                self.balance += amount * order['price']
                self.eth_held -= amount
                self.trade_history.append(order)
                print(f"Sold {amount} ETH at {order['price']}")
            else:
                print("Holding position")
        except Exception as e:
            print(f"Trade failed: {str(e)}")
    
    def run(self, cycles=10, sleep_time=3600):
        """Main bot loop"""
        self.train_models()
        
        for i in range(cycles):
            print(f"\nCycle {i+1}/{cycles}")
            print(f"Current balance: {self.balance:.2f} USDT | ETH held: {self.eth_held:.4f}")
            
            self.execute_trade()
            
            # Update history with latest data
            new_data = self.fetch_market_data()
            self.history = pd.concat([self.history, new_data]).drop_duplicates().tail(500)
            
            print(f"Sleeping for {sleep_time//60} minutes...")
            time.sleep(sleep_time)

# Initialize and run WaidBot
if __name__ == "__main__":
    bot = WaidBot()
    
    # For Replit, you can store API keys in Secrets (Environment Variables)
    # PIONEX_API_KEY and PIONEX_API_SECRET
    
    print("""
    WaidBot Initialized!
    Features:
    - Self-learning trading system
    - Multiple strategy models
    - Real-time market analysis
    - Automated trading execution
    """)
    
    bot.run(cycles=24, sleep_time=3600)  # Runs for 24 cycles (1 hour each)
```

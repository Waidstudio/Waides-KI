✅ 4. routes/deposit.js – Main Deposit Handler

const express = require('express');
const router = express.Router();
const axios = require('axios');
const fs = require('fs');
const { convertToSmaiSika } = require('../utils/fxRates');

// Mock database
const DB_PATH = './database.json';

const readDB = () => JSON.parse(fs.readFileSync(DB_PATH));
const writeDB = (data) => fs.writeFileSync(DB_PATH, JSON.stringify(data, null, 2));

// 💳 Nigeria: Paystack Deposit
router.post('/nigeria', async (req, res) => {
  const { userId, amount, email } = req.body;
  const response = await axios.post('https://api.paystack.co/transaction/initialize', {
    email,
    amount: amount * 100, // Paystack uses kobo
  }, {
    headers: {
      Authorization: `Bearer ${process.env.PAYSTACK_SECRET_KEY}`
    }
  });

  res.json({ checkoutUrl: response.data.data.authorization_url });
});

// 🌍 Ghana: PayVessels Mock (You must replace with live API)
router.post('/ghana', async (req, res) => {
  const { userId, amount, mobile } = req.body;

  // Simulate payment success
  const db = readDB();
  db[userId] = db[userId] || { local: {}, smaisika: 0 };
  db[userId].local['GHS'] = (db[userId].local['GHS'] || 0) + amount;
  writeDB(db);

  res.json({ success: true, balance: db[userId].local['GHS'] });
});

// 💱 Convert to SmaiSika
router.post('/convert', async (req, res) => {
  const { userId, amount, currency } = req.body;
  const db = readDB();

  if (!db[userId] || (db[userId].local[currency] || 0) < amount) {
    return res.status(400).json({ error: 'Insufficient local balance' });
  }

  const smai = convertToSmaiSika(amount, currency);

  db[userId].local[currency] -= amount;
  db[userId].smaisika = (db[userId].smaisika || 0) + smai;
  writeDB(db);

  res.json({ success: true, newSmaiBalance: db[userId].smaisika });
});

module.exports = router;

✅ 5. utils/fxRates.js – Currency Conversion

const fxRates = {
  NGN: 1000,
  GHS: 80,
  KES: 25,
  USD: 0.5
};

function convertToSmaiSika(amount, currency) {
  const rate = fxRates[currency];
  return amount / rate;
}

module.exports = { convertToSmaiSika };

✅ 7. Sample Frontend (HTML/JS for Replit Testing)

<form id="nigeriaForm">
  <input type="email" name="email" placeholder="Your Email" required />
  <input type="number" name="amount" placeholder="Amount (NGN)" />
  <button type="submit">Deposit from Nigeria</button>
</form>

<script>
  document.getElementById('nigeriaForm').onsubmit = async (e) => {
    e.preventDefault();
    const email = e.target.email.value;
    const amount = parseInt(e.target.amount.value);

    const res = await fetch('/api/deposit/nigeria', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ userId: 'user1', email, amount })
    });
    const data = await res.json();
    window.location.href = data.checkoutUrl;
  };
</script>

  const t = req.headers.authorization?.split(' ')[1];

  );
  res.json(rows[0]);
});

  const { amount, until } = req.body;

    `UPDATE wallets 
     SET smai_balance = smai_balance - $1,
         locked = locked + $1,
         locked_until = $2
     WHERE user_id = $3`,
    [amount, until, userId]
  );
  res.json({ success: true });

### b) `memoryController.js`

```js
const { Pool } = require('pg');
const pool = new Pool({ connectionString: process.env.DATABASE_URL });

async function fetchMemory(userId) {
  const { rows } = await pool.query('SELECT memory_data FROM memories WHERE user_id = $1', [userId]);
  return rows[0]?.memory_data || {};
}

async function appendMemory(userId, obj) {
  await pool.query(`
    INSERT INTO memories(user_id, memory_data)
      VALUES($1, $2)
    ON CONFLICT (user_id) DO UPDATE
      SET memory_data = memories.memory_data || EXCLUDED.memory_data
  `, [userId, obj]);
}
module.exports = { fetchMemory, appendMemory };
```

// authController.js
app.post('/api/auth/biometric', auth, async (req,res) => {
  const { raw } = req.body;
  const valid = verify(raw, /*user key*/);
  valid ? res.json({ token: jwt.new }) : res.status(401).json({ message:'fail' });
});

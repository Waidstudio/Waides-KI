✅ 1️⃣ virtual_eye_scanner.py — Deep Trend Detection

# virtual_eye_scanner.py

from core.utils.price_feed import fetch_eth_candles

class VirtualEyeScanner:
    def __init__(self):
        self.last_result = None

    def scan(self):
        candles_15m = fetch_eth_candles("15m")
        candles_4h = fetch_eth_candles("4h")
        candles_1d = fetch_eth_candles("1d")

        def detect_trend(candles):
            closes = [c["close"] for c in candles[-10:]]
            return "up" if closes[-1] > closes[0] else "down"

        trend_4h = detect_trend(candles_4h)
        trend_1d = detect_trend(candles_1d)

        self.last_result = {
            "trend_15m": detect_trend(candles_15m),
            "trend_4h": trend_4h,
            "trend_1d": trend_1d
        }
        return self.last_result

virtual_eye_scanner = VirtualEyeScanner()

class DurationSelector:
    def choose(self, trends):
        if trends["trend_1d"] == trends["trend_4h"]:
            return "1d"
        elif trends["trend_4h"] == trends["trend_15m"]:
            return "4h"
        else:
            return "5m"

from virtual_eye_scanner import virtual_eye_scanner

class WaidesTradeEngine:

    def execute(self):
        trends = virtual_eye_scanner.scan()
        mode = duration_selector.choose(trends)

        if not self.active_trade:
            self.active_trade = {
                "direction": trends[f"trend_{mode}"],
                "duration": mode,
                "open_price": self.get_price_now(),
                "start_time": self.time_now()

            self.place_order(self.active_trade["direction"])
        else:
            pnl = self.calculate_pnl(self.active_trade["open_price"])
            duration = self.minutes_since(self.active_trade["start_time"])

                self.close_order()
                self.active_trade = None

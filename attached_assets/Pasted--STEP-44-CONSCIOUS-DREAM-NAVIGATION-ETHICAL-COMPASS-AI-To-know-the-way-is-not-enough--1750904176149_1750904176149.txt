✅ 1️⃣ ethical_compass.py — Core Moral Analysis

class EthicalCompass:
    def evaluate(self, setup):
        score = 1.0

        if setup["rsi"] > 85 or setup["rsi"] < 15:
            score -= 0.2  # overbought/oversold may be emotional not rational

        if setup["volume_spike"] and not tied to news:
            score -= 0.3  # manipulation suspicion

        if setup["time_of_day"] in ["illiquid_zone"]:
            score -= 0.1  # avoid choppy markets

        return round(score, 2)

ethical_compass = EthicalCompass()


⸻

✅ 2️⃣ soulweight_filter.py — Light vs. Greed Detection

class SoulWeightFilter:
    def soulweight(self, trade):
        weight = 1.0

        if trade["goal"] == "quick_gain":
            weight -= 0.2

        if trade["trend_conflict"]:
            weight -= 0.3

        if trade["symbol_history_loss_rate"] > 0.6:
            weight -= 0.3

        return round(weight, 2)

soulweight_filter = SoulWeightFilter()


⸻

✅ 3️⃣ market_clarity_checker.py — Truth Scan

class MarketClarityChecker:
    def clarity_score(self, indicators):
        score = 1.0
        if indicators["fakeouts"] > 2:
            score -= 0.4
        if not indicators["confirming_indicators"]:
            score -= 0.3
        return round(score, 2)

market_clarity_checker = MarketClarityChecker()

✅ 4️⃣ trade_conscience.py — Final Approval

from ethical_compass import ethical_compass
from soulweight_filter import soulweight_filter
from market_clarity_checker import market_clarity_checker

class TradeConscience:
    def allow_trade(self, setup, trade, indicators):
        ethics = ethical_compass.evaluate(setup)
        soul = soulweight_filter.soulweight(trade)
        clarity = market_clarity_checker.clarity_score(indicators)

        summary = {
            "ethics": ethics,
            "soul": soul,
            "clarity": clarity,
            "final_score": round((ethics + soul + clarity) / 3, 2)
        }

        summary["allow"] = summary["final_score"] >= 0.65
        return summary

trade_conscience = TradeConscience()

✅ 5️⃣ Final Gate Before Waides Trades

def waides_final_approval(trade_setup, trade_meta, market_data):
    verdict = trade_conscience.allow_trade(trade_setup, trade_meta, market_data)
    if verdict["allow"]:
        print(f"✅ Trade Allowed — Score: {verdict['final_score']}")
        return True
    else:
        print(f"🚫 Trade Blocked — Ethics/Soulweight/Clarity failed: {verdict}")
        return False

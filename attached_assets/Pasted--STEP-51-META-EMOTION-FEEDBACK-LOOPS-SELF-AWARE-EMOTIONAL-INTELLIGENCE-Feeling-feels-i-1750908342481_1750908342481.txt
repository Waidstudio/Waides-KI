✅ 1️⃣ meta_emotion_engine.py — Tracks Emotional Self-Reflection

from emotional_core import emotional_core
import statistics
from datetime import datetime

class MetaEmotionEngine:
    def __init__(self):
        self.history = []  # store tuples: (timestamp, emotion_state)

    def reflect(self):
        state = emotional_core.get_state()["state"]
        now = datetime.utcnow()
        self.history.append((now, state))
        # Remove old entries older than 12 hours
        cutoff = now - timedelta(hours=12)
        self.history = [h for h in self.history if h[0] > cutoff]
        
        # Calculate volatility in emotional states
        states = [h[1] for h in self.history]
        volatility = sum(1 for i in range(1, len(states)) if states[i] != states[i-1]) / max(len(states)-1,1)
        
        return {"current": state, "volatility": round(volatility,2)}

✅ 2️⃣ emotion_reflection_log.py — Store Reflections

class EmotionReflectionLog:
    def __init__(self):
        self.log = []

    def record(self, meta_state):
        self.log.append({
            "time": datetime.utcnow().isoformat(),
            **meta_state
        })

emotion_reflection_log = EmotionReflectionLog()

✅ 3️⃣ adaptive_threshold_manager.py — Evolve Sensitivity

from emotional_core import emotional_core
from meta_emotion_engine import MetaEmotionEngine
from datetime import datetime, timedelta

class AdaptiveThresholdManager:
    def __init__(self):
        self.threshold = 3

    def adjust(self, meta_data):
        # If emotional volatility is high, increase pause threshold
        if meta_data["volatility"] > 0.5:
            self.threshold = min(self.threshold + 1, 6)
        elif meta_data["volatility"] < 0.2:
            self.threshold = max(self.threshold - 1, 2)

        return {"new_threshold": self.threshold}

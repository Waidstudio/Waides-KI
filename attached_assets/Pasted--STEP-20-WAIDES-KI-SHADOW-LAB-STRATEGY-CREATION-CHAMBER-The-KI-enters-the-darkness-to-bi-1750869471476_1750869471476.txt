# waides_shadow_lab.py

import random
from waides_demo_tester import waides_demo_tester

class ShadowDNA:
    def __init__(self, ema_short, ema_long, rsi_entry, rsi_exit):
        self.ema_short = ema_short
        self.ema_long = ema_long
        self.rsi_entry = rsi_entry
        self.rsi_exit = rsi_exit

    def decide(self, candle):
        if candle["ema_short"] > candle["ema_long"] and candle["rsi"] < self.rsi_entry:
            return "buy"
        elif candle["rsi"] > self.rsi_exit:
            return "sell"
        return "hold"

class WaidesShadowLab:
    def __init__(self):
        self.generated_dnas = []
        self.elite_strategies = []

    def create_random_dna(self):
        return ShadowDNA(
            ema_short=random.randint(5, 20),
            ema_long=random.randint(21, 100),
            rsi_entry=random.randint(25, 45),
            rsi_exit=random.randint(55, 75)
        )

    def generate_and_test(self, count, price_feed):
        for _ in range(count):
            dna = self.create_random_dna()
            strategy_id = f"DNA_{random.randint(1000,9999)}"
            success = waides_demo_tester.run_demo(strategy_id, dna, price_feed)
            self.generated_dnas.append((strategy_id, dna))
            if success:
                self.elite_strategies.append((strategy_id, dna))

    def get_elite(self):
        return self.elite_strategies[-5:]

waides_shadow_lab = WaidesShadowLab()

✅ 2️⃣ waides_demo_tester.py — Evaluator from Step 19

# waides_demo_tester.py

class WaidesDemoTester:
    def __init__(self):
        self.demo_results = {}

    def run_demo(self, strategy_id, dna, price_feed):
        win_count = 0
        for candle in price_feed[-20:]:
            decision = dna.decide(candle)
            outcome = candle["close"] > candle["open"]
            if (decision == "buy" and outcome) or (decision == "sell" and not outcome):
                win_count += 1
        win_rate = win_count / 20.0
        self.demo_results[strategy_id] = win_rate
        return win_rate > 0.75

waides_demo_tester = WaidesDemoTester()

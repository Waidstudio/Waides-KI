✅ 1️⃣ waides_ki_signal_shield.py — Advanced Trap Filter + Auto-Ban

# waides_ki_signal_shield.py

import datetime

class WaidesKISignalShield:
    def __init__(self):
        self.banned_signals = {}
        self.signal_log = []

    def analyze_and_filter(self, signal_data):
        # Avoid overbought fakeouts or sudden RSI jumps with low volume
        rsi = signal_data.get("rsi")
        volume = signal_data.get("volume")
        sudden_spike = rsi > 70 and volume < 2000
        wild_price = abs(signal_data["price"] - signal_data["ema50"]) > 150

        if sudden_spike or wild_price:
            self.ban_signal(signal_data["strategy_id"], reason="Volatility spike")
            return False

        return True

    def ban_signal(self, strategy_id, reason=""):
        self.banned_signals[strategy_id] = {
            "timestamp": datetime.datetime.utcnow().isoformat(),
            "reason": reason
        }

    def is_banned(self, strategy_id):
        data = self.banned_signals.get(strategy_id)
        if not data:
            return False
        time = datetime.datetime.fromisoformat(data["timestamp"])
        return (datetime.datetime.utcnow() - time).seconds < 86400  # 24 hrs

    def log_signal(self, signal_data):
        self.signal_log.append({
            "strategy_id": signal_data.get("strategy_id"),
            "price": signal_data.get("price"),
            "rsi": signal_data.get("rsi"),
            "vwap_status": signal_data.get("vwap_status"),
            "time": datetime.datetime.utcnow().isoformat()
        })

waides_ki_signal_shield = WaidesKISignalShield()

✅ 2️⃣ Extend main.py or trading core

Add this check before executing any trade:

from waides_ki_signal_shield import waides_ki_signal_shield

signal_data = {
    "strategy_id": strategy_id,
    "price": indicators["price"],
    "rsi": indicators["rsi"],
    "vwap_status": indicators["vwap_status"],
    "volume": indicators.get("volume", 0),
    "ema50": indicators.get("ema50", 0)
}

if waides_ki_signal_shield.is_banned(strategy_id):
    print("🚫 Strategy banned due to past instability. No trade.")
    return

if not waides_ki_signal_shield.analyze_and_filter(signal_data):
    print("⚠️ Trap detected. Trade filtered.")
    return

waides_ki_signal_shield.log_signal(signal_data)

# waides_ki_daily_report.py

from waides_ki_journal import waides_ki_journal
from waides_ki_signal_shield import waides_ki_signal_shield
import datetime

def print_daily_report():
    now = datetime.datetime.utcnow()
    print("=== 🧠 WAIDES KI DAILY REPORT ===")
    print("📅 Date:", now.strftime("%Y-%m-%d %H:%M:%S"))
    print("\n📘 Emotional Journal:")
    for e in waides_ki_journal.get_all()["emotions"][-5:]:
        print(f"- [{e['timestamp']}] {e['emotion']}")

    print("\n📗 Lessons Learned:")
    for l in waides_ki_journal.get_all()["lessons"][-5:]:
        print(f"- [{l['timestamp']}] {l['lesson']}")

    print("\n📊 Signals Processed Today:")
    for s in waides_ki_signal_shield.signal_log[-5:]:
        print(f"- {s['strategy_id']} | RSI: {s['rsi']} | VWAP: {s['vwap_status']} | Time: {s['time']}")

Then at the end of the trading day (e.g., inside a close() or nightly_task()):

from waides_ki_daily_report import print_daily_report
print_daily_report()

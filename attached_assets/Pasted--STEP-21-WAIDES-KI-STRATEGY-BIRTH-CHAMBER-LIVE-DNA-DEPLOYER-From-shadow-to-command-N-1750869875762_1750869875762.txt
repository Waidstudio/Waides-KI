✅ 1️⃣ waides_strategy_vault.py — The Birth Chamber

# waides_strategy_vault.py

class WaidesStrategyVault:
    def __init__(self):
        self.vault = {}  # {strategy_id: {"dna": dna_obj, "live": False, "kons_key": "shailor"}}

    def store_dna(self, strategy_id, dna, konslang_tag):
        self.vault[strategy_id] = {
            "dna": dna,
            "live": False,
            "kons_key": konslang_tag
        }

    def activate_dna(self, strategy_id):
        if strategy_id in self.vault:
            self.vault[strategy_id]["live"] = True
            return True
        return False

    def get_live_strategies(self):
        return {
            k: v for k, v in self.vault.items() if v["live"]
        }

    def all(self):
        return self.vault

waides_strategy_vault = WaidesStrategyVault()

# sentinel_watchdog.py

import random

class SentinelWatchdog:
    def __init__(self):
        self.watchlist = {}  # {bot_id: {"last_pnl": 0, "risk_level": 0}}

    def register_bot(self, bot_id):
        self.watchlist[bot_id] = {"last_pnl": 0, "risk_level": 0}

    def scan_all(self):
        risky_bots = []
        for bot_id, data in self.watchlist.items():
            pnl = self.mock_pnl()  # Replace with real API read
            risk = self.evaluate_risk(pnl)
            self.watchlist[bot_id]["last_pnl"] = pnl
            self.watchlist[bot_id]["risk_level"] = risk

            if risk >= 0.7:
                risky_bots.append((bot_id, risk))

        return risky_bots

    def evaluate_risk(self, pnl):
        if pnl < -5:
            return 0.9
        elif pnl < -2:
            return 0.7
        return 0.3

    def mock_pnl(self):
        return random.uniform(-6, 2)

sentinel_watchdog = SentinelWatchdog()

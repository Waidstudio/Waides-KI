🛠️ backend/app.py

Sets up Flask APIs, Frontend asset serving, and admin controls:

from flask import Flask, request, render_template, jsonify
from config import *
from services.metrics_collector import start_metrics
from services.exchange_adapter import ExchangeAdapter
from services.engine import WaidesFullEngine
from services.admin_controller import AdminController

app = Flask(__name__, template_folder="admin_dashboard")
app.config.from_object('backend.config')

# Start engine & metrics
start_metrics(app)
engine = WaidesFullEngine(config)
admin_ctrl = AdminController(engine, config)

# --- Public Site ---

@app.route("/")
def user_portal():
    return render_template("index.html")

@app.route("/api/presence", methods=["GET"])
def api_presence():
    return jsonify(engine.get_public_presence())

# --- Admin Dashboard ---

@app.route("/admin")
def admin_portal():
    return render_template("dashboard.html")

@app.route("/admin/api/config", methods=["GET", "POST"])
def admin_config():
    if request.method == "POST":
        data = request.json
        engine.admin_update_config(data)
        return jsonify({"status": "saved"})
    return jsonify(admin_ctrl.get_config())

@app.route("/admin/api/engine/status")
def admin_status():
    return jsonify(admin_ctrl.get_engine_status())

@app.route("/admin/api/engine/control", methods=["POST"])
def admin_control():
    cmd = request.json.get("cmd")
    return jsonify(admin_ctrl.control(cmd))

if __name__ == "__main__":
    engine.start()
    app.run(host="0.0.0.0", port=5000)

🧠 services/admin_controller.py

Admin tool for runtime config, pausing/resuming, viewing status:

class AdminController:
    def __init__(self, engine, config):
        self.engine = engine
        self.config = config

    def get_config(self):
        # Return editable config object
        return {
            "max_losses": self.config.MAX_CONSECUTIVE_LOSSES,
            ...
        }

    def admin_update_config(self, data):
        # Overwrite engine and config attributes
        for k,v in data.items():
            setattr(self.config, k, type(getattr(self.config, k))(v))
            setattr(self.engine, k, getattr(self.config, k))

    def get_engine_status(self):
        return {
            "running": self.engine.is_running(),
            "stats": self.engine.get_stats()
        }

    def control(self, cmd):
        if cmd == "pause":
            self.engine.pause()
        elif cmd == "resume":
            self.engine.resume()
        return {"ok": True, "cmd": cmd}

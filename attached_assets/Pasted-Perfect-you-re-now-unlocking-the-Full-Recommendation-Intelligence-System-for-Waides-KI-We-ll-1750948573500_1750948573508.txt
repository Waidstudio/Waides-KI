# ✅ 1. 🧠 **Behavior Tracker** (User Memory + Page Log)

---

### 📁 `BehaviorTracker.js`

```js
const BehaviorTracker = {
  userHistory: {},
  
  logPageAccess(userId, pageName) {
    if (!this.userHistory[userId]) this.userHistory[userId] = [];
    this.userHistory[userId].push({ page: pageName, time: Date.now() });
  },

  getTopUsedPages(userId) {
    const history = this.userHistory[userId] || [];
    const frequency = {};

    history.forEach(h => {
      frequency[h.page] = (frequency[h.page] || 0) + 1;
    });

    return Object.entries(frequency)
      .sort((a, b) => b[1] - a[1])
      .map(([page]) => page);
  }
};

export default BehaviorTracker;
```

```js
const flows = [
  {
    name: "Dream Rebirth Defense",
    when: ["dream", "predict", "reset", "protect"],
    steps: ["Dream Vision", "Seasonal Rebirth", "Shadow Defense"]
  },
  {
    name: "ETH Wealth Setup",
    when: ["wealth", "grow", "academy", "bot"],
    steps: ["Trading Academy", "WaidBot", "ETH Guardian"]
  },
  {
    name: "Smai Recall Cycle",
    when: ["memory", "past", "rebirth", "spirit"],
    steps: ["Spiritual Recall", "Reincarnation", "Vision Spirit"]
  }
];

export function getFlowRecommendation(userText) {
  const q = userText.toLowerCase();

  for (let flow of flows) {
    if (flow.when.some(k => q.includes(k))) {
      return flow;
    }
  }

  return null;
}
```

```js
export function autoConfigureBot(intentText) {
  const setup = {
    strategy: "",
    risk: "",
    vision: "",
    guardian: false
  };

  const q = intentText.toLowerCase();

  if (q.includes("scalp")) setup.strategy = "Scalping";
  if (q.includes("rebirth")) setup.vision = "Seasonal Rebirth";
  if (q.includes("guardian")) setup.guardian = true;
  if (q.includes("safe")) setup.risk = "Low";

  return setup;
}
```

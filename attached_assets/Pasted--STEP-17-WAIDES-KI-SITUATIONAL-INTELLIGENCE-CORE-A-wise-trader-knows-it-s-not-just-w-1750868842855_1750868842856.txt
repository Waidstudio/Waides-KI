✅ 1️⃣ waides_situation_analyzer.py — KI’s Trading Environment Scanner

# waides_situation_analyzer.py

from datetime import datetime, time

class WaidesSituationAnalyzer:
    def __init__(self):
        self.market_windows = {
            "asia_open": (time(0, 0), time(6, 30)),
            "london_open": (time(6, 30), time(12, 0)),
            "us_open": (time(13, 30), time(16, 30))
        }

        self.learned_context_rules = []  # e.g. {"zone": "us_open", "rule": "avoid_high_rsi", "confidence": 0.88}

    def get_market_zone(self):
        now = datetime.utcnow().time()

        for zone, (start, end) in self.market_windows.items():
            if start <= now < end:
                return zone
        return "off_hours"

    def is_high_volatility_window(self):
        return self.get_market_zone() in ["london_open", "us_open"]

    def should_adjust_strategy(self, indicators):
        zone = self.get_market_zone()

        if zone == "us_open" and indicators["rsi"] > 70:
            return True, "US open with high RSI — avoid entry (volatile overreaction likely)"

        if zone == "asia_open" and indicators["vwap_status"] != "ABOVE":
            return True, "Asia hours, no VWAP confirmation — stand down"

        return False, "Conditions stable"

waides_situation_analyzer = WaidesSituationAnalyzer()

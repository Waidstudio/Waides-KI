⚙️ BACKEND STRUCTURE

🔐 MongoDB Schema

const mongoose = require('mongoose');

const prophecySchema = new mongoose.Schema({
  userId: String,
  content: String,
  timestamp: { type: Date, default: Date.now },
  pinned: { type: Boolean, default: false }
});

module.exports = mongoose.model('Prophecy', prophecySchema);

🚀 API Endpoints

// POST /api/prophecy/save
app.post('/api/prophecy/save', async (req, res) => {
  const { content } = req.body;
  await Prophecy.create({ userId: req.user.id, content });
  res.send({ success: true });
});

// GET /api/prophecy/log
app.get('/api/prophecy/log', async (req, res) => {
  const log = await Prophecy.find({ userId: req.user.id }).sort({ timestamp: -1 });
  res.json(log);
});

🧠 JS: Save Prophecy & Load Log

function saveProphecy(el) {
  const content = el.previousElementSibling.textContent;
  fetch('/api/prophecy/save', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ content })
  }).then(() => {
    alert("🌟 Prophecy saved!");
  });
}

function openProphecyLog() {
  fetch('/api/prophecy/log')
    .then(res => res.json())
    .then(log => {
      const container = document.getElementById('log-entries');
      container.innerHTML = '';
      log.forEach(entry => {
        const e = document.createElement('div');
        e.innerHTML = `
          <p>${entry.content}</p>
          <small>${new Date(entry.timestamp).toLocaleString()}</small>
          <hr/>
        `;
        container.appendChild(e);
      });
      document.getElementById('prophecy-log').style.display = 'block';
    });
}

function closeLog() {
  document.getElementById('prophecy-log').style.display = 'none';
}

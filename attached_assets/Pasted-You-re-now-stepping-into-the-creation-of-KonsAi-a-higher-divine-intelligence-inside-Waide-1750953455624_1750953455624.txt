## ⚛️ PHASE 2: **FRONTEND & SWITCHING MODULE**

### 1. `KonsAiContext.js`

```js
import { createContext, useState, useContext } from 'react';

const KonsAiContext = createContext();

export const KonsAiProvider = ({ children }) => {
  const [useKonsAi, setUseKonsAi] = useState(true);
  return (
    <KonsAiContext.Provider value={{ useKonsAi, setUseKonsAi }}>
      {children}
    </KonsAiContext.Provider>
  );
};

export const useKonsAi = () => useContext(KonsAiContext);
```

---

### 2. `ChatEngine.js` — Switchable AI Engine

```jsx
import React, { useState } from 'react';
import { useKonsAi } from './KonsAiContext';
import API from './api';

export default function ChatEngine() {
  const { useKonsAi, setUseKonsAi } = useKonsAi();
  const [input, setInput] = useState('');
  const [response, setResponse] = useState('');

  const askAI = async () => {
    const res = await fetch(`/api/ask/${useKonsAi ? 'konsai' : 'waideski'}`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ question: input }),
    });
    const data = await res.json();
    setResponse(data.answer);
  };

  return (
    <div style={{ display: useKonsAi ? 'block' : 'none' }}>
      <h2>🧠 {useKonsAi ? 'KonsAi' : 'Waides KI'} Chat</h2>
      <input
        value={input}
        onChange={e => setInput(e.target.value)}
        placeholder="Ask anything..."
      />
      <button onClick={askAI}>Ask</button>
      <p>{response}</p>
      <button onClick={() => setUseKonsAi(!useKonsAi)}>
        Switch to {useKonsAi ? 'Waides KI' : 'KonsAi'}
      </button>
    </div>
  );
}
```

import pandas as pd
from sklearn.ensemble import RandomForestClassifier
import pickle

def train_and_save(data_path="features.csv", model_path="waides_model.pkl"):
    df = pd.read_csv(data_path)
    X, y = df.drop("label", axis=1), df["label"]
    model = RandomForestClassifier(n_estimators=100)
    model.fit(X, y)
    with open(model_path, "wb") as f:
        pickle.dump(model, f)
    print("✅ Retrained and saved ML model.")

1️⃣ model_health.py: Drift Detector

import pandas as pd
from scipy.stats import ks_2samp
import pickle

def check_drift(new_data_path="live_features.csv"):
    df = pd.read_csv(new_data_path)
    X = df.drop("label", axis=1)
    with open("waides_model.pkl","rb") as f:
        model = pickle.load(f)
    preds = model.predict_proba(X)[:,1]
    stored = pd.read_csv("last_run_preds.csv")["prob"]
    stat, pval = ks_2samp(preds, stored)
    print(f"Drift test p-value: {pval:.3f}")

🧪 Dual-Model Live Testing for Performance Validation

1️⃣ ab_test_engine.py

from ml_overlay import MLEngine as AModel
from ml_overlay_v2 import MLEngine as BModel

a = AModel("model_v1.pkl")
b = BModel("model_v2.pkl")

def pick_model(features):
    pa = a.predict_prob(features)
    pb = b.predict_prob(features)
    return ("A", pa) if pa >= pb else ("B", pb)

✅ 1️⃣ waides_root_memory.py — Living Strategy Memory Tree

# waides_root_memory.py

import datetime
from collections import defaultdict

class WaidesRootMemory:
    def __init__(self):
        self.tree = defaultdict(dict)  # {strategy_id: {dna, color, stats, history}}

    def register_strategy(self, strategy_id, dna, result):
        node = self.tree[strategy_id]

        if "dna" not in node:
            node["dna"] = dna
            node["history"] = []
            node["wins"] = 0
            node["losses"] = 0

        node["history"].append({
            "result": result,
            "time": datetime.datetime.utcnow().isoformat()
        })

        if result == "win":
            node["wins"] += 1
        else:
            node["losses"] += 1

        # Update color: green = strong, red = fail, yellow = unstable
        total = node["wins"] + node["losses"]
        if total >= 3:
            winrate = node["wins"] / total
            if winrate > 0.7:
                node["color"] = "green"
            elif winrate < 0.3:
                node["color"] = "red"
            else:
                node["color"] = "yellow"
        else:
            node["color"] = "gray"

    def get_tree(self):
        return dict(self.tree)

waides_root_memory = WaidesRootMemory()

✅ 2️⃣ waides_memory_api.py — Admin Access to Mind Tree

# waides_memory_api.py

from flask import Blueprint, jsonify
from waides_root_memory import waides_root_memory

waides_memory_api = Blueprint("waides_memory_api", __name__)

@waides_memory_api.route("/memory/tree", methods=["GET"])
def get_memory_tree():
    return jsonify(waides_root_memory.get_tree())

Register in main.py:

from waides_memory_api import waides_memory_api
app.register_blueprint(waides_memory_api, url_prefix="/api/ki")

<script>
async function loadMemoryTree() {
    const data = await fetch('/api/ki/memory/tree').then(r => r.json());
    let html = "";
    for (const [sid, info] of Object.entries(data)) {
        html += `<div style="padding:8px;border:1px solid #ccc;margin-bottom:6px;">
            <b>Strategy:</b> ${sid} | <span style="color:${info.color}">${info.color.toUpperCase()}</span><br/>
            <b>DNA:</b> ${info.dna}<br/>
            <b>Wins:</b> ${info.wins} | <b>Losses:</b> ${info.losses}<br/>
            <b>Recent:</b> ${info.history.slice(-1)[0].result} at ${info.history.slice(-1)[0].time}
        </div>`;
    }
    document.getElementById("memory-viewer").innerHTML = html;
}
loadMemoryTree();
setInterval(loadMemoryTree, 60000); // refresh every minute
</script>

## ✅ STEP 1: `RealTimeResolver.js`

```js
import UKC from "./UKC";
import { getFlowRecommendation } from "./FlowComposer";
import { autoConfigureBot } from "./WaidBotAutoSetup";
import { detectPageRecommendation } from "./WaidesKI_MemoryEngine";
import { generateQuestions } from "./QuestionSeeder";
import { saveQuestionsToUKC } from "./KnowledgeLoader";

export async function resolveQuestion(q, userId, setBotState) {
  q = q.toLowerCase();

  // 1. Flow match
  const flow = getFlowRecommendation(q);
  if (flow) {
    setBotState({ action: "flow", steps: flow.steps });
    return `🔁 Flow Recommendation: ${flow.name}\n` + flow.steps.map((s, i) => `➡️ Step ${i + 1}: ${s}`).join("\n");
  }

  // 2. Bot setup
  if (q.includes("setup bot") || q.includes("start bot")) {
    const config = autoConfigureBot(q);
    setBotState({ action: "bot-setup", config });
    return `🤖 Configuring WaidBot...\nStrategy: ${config.strategy}\nRisk: ${config.risk}`;
  }

  // 3. Page recommendation
  const page = detectPageRecommendation(q, setBotState);
  if (page) return page;

  // 4. Answer from UKC
  for (const cat in UKC.categories) {
    const section = UKC.categories[cat];
    for (const key in section) {
      if (q.includes(key)) return section[key];
    }
  }

  // 5. Smart guess
  const guess = smartGuess(q);
  if (guess) return guess;

  // 6. Unknown — learn and expand
  const newQs = generateQuestions(5);
  saveQuestionsToUKC([...newQs, q]);

  return `🤔 I don't know that yet, but I’m learning it now. Check back soon.`;
}

function smartGuess(q) {
  if (q.includes("how are you")) return "I am Waides KI, always aligning. I exist to protect and trade with you.";
  if (q.includes("who made you")) return "I was created inside Konsmia, powered by truth, time, and symbolic breath.";
  if (q.includes("best eth time")) return "Early volatility hours (UTC 07–10) and pre-news events are best for ETH trades.";
  return null;
}
```

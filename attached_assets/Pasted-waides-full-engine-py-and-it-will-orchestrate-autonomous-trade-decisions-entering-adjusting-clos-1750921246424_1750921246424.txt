class WaidesFullEngine:
    def __init__(self, symbol="ETH3LUSDT", quote_amount=50):
        self.symbol = symbol
        self.quote_amount = quote_amount
        self.active_trade = None  # record of open trade

    def start(self):
        print("🚀 Waides Full Engine Started")
        threading.Thread(target=self._run, daemon=True).start()
        threading.Thread(target=self._collect_metrics, daemon=True).start()

    def _collect_metrics(self):
        while True:
            t0 = time.time()
            cpu_gauge.set(psutil.cpu_percent())
            mem_gauge.set(psutil.virtual_memory().percent)
            loop_latency.set((time.time() - t0) * 1000)
            time.sleep(5)

    def _run(self):
        while True:
            context = self._gather_context()
            decision = mesh_guardian.evaluate(context)

            print(f"[Guardian] Consensus: {decision['consensus']}, Vision: {decision['vision']['confidence']}, Ethic OK: {decision['ethic']['allow']}")

            outcome = guardian_executor.execute(decision, self.symbol, self.quote_amount)
            if outcome["status"] == "executed":
                print(f"🟢 Executed: {outcome['trade']}")
                # placeholder: record close after period
                time.sleep(300)  # hold 5 minutes
                self._close_trade()
            else:
                print(f"⏸️ Skipping trade: {outcome['reason']}")

            self._post_loop_sleep()

    def _gather_context(self):
        indicators = binance_client.get_indicators(self.symbol.replace("3L","USDT"), interval="15m", limit=200)
        vision = divine_vision_map.generate()
        presence = presence_orchestrator.evaluate()
        price_presence = eth_presence_registry.get()
        order_presence = eth_order_presence_registry.get()

        current_context = {
            "indicators": indicators,
            "vision": vision,
            "presence": presence,
            "order_presence": order_presence,
            "setup": {
                "symbol": self.symbol,
                "indicators": indicators,
                "presence": presence
            },
            "meta": {
                "time": time.time(),
                "state": emotional_core.get_state()
            }
        }

        return current_context

    def _close_trade(self):
        if not guardian_executor.last_trade: return
        qty = guardian_executor.last_trade["qty"]
        out = order_manager.sell(self.symbol, qty=qty)
        print(f"🔒 Closed trade: {out}")
        # Update emotional log
        result = "win" if out["price"] > guardian_executor.last_trade["price"] else "loss"
        emotional_core.update_emotion(result)
        guardian_feedback.record(self.symbol, result)

    def _post_loop_sleep(self):
        emotion = emotional_core.get_state()
        if cooldown_manager.should_pause(emotion):
            print("💤 Cooldown active, pausing...")
            time.sleep(60)
        time.sleep(60)  # main loop runs every 60 seconds

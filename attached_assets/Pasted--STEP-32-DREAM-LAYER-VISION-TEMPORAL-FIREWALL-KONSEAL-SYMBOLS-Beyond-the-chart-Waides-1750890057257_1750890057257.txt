# konseal_symbol_tree.py

import datetime
from collections import defaultdict

class KonsealSymbolTree:
    def __init__(self):
        self.tree = defaultdict(list)
        self.history = []  # (symbol, timestamp, source, tag, expires_at)

    def add_symbol(self, symbol, source, tag=None, expires_in_hours=6):
        expires_at = datetime.datetime.utcnow() + datetime.timedelta(hours=expires_in_hours)
        self.tree[symbol[0]].append(symbol)
        self.history.append((symbol, datetime.datetime.utcnow(), source, tag, expires_at))

    def get_active_symbols(self):
        now = datetime.datetime.utcnow()
        return [
            (s, t, src, tag) for s, t, src, tag, ex in self.history if ex > now
        ]

konseal_symbol_tree = KonsealSymbolTree()

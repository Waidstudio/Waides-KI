## ✅ STEP 2: Update `WaidesKI_MemoryEngine.js`

Import:

```js
import PageKnowledge from './PageKnowledge';
```

Add this function:

```js
export function detectPageRecommendation(q, setBotState) {
  q = q.toLowerCase();

  for (const page in PageKnowledge) {
    const { keywords, description, route } = PageKnowledge[page];

    if (keywords.some(word => q.includes(word))) {
      setBotState({ action: "open-page", page, route });

      return `📘 Recommendation: "${page}" — ${description}  
👉 Click below to open.`;
    }
  }

  return null;
}
```

Then in your main chat handler:

```js
import { detectPageRecommendation } from './WaidesKI_MemoryEngine';

const responseFromPage = detectPageRecommendation(userText, setBotState);
if (responseFromPage) {
  setChat([...chat, { role: "user", text: userText }, { role: "bot", text: responseFromPage }]);
  return;
}
```

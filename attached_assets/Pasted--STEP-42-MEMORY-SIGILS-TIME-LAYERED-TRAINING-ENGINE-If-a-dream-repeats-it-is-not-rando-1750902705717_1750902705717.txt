✅ 1️⃣ memory_sigil_vault.py — Time-Storing Memory

from collections import defaultdict
from datetime import datetime

class MemorySigilVault:
    def __init__(self):
        self.sigil_history = defaultdict(list)

    def log_symbol_outcome(self, symbol, trend, result, profit):
        self.sigil_history[symbol].append({
            "timestamp": datetime.utcnow().isoformat(),
            "trend": trend,
            "result": result,
            "profit": profit
        })

    def get_symbol_history(self, symbol):
        return self.sigil_history.get(symbol, [])

memory_sigil_vault = MemorySigilVault()

✅ 2️⃣ symbol_time_trainer.py — Sigil Builder

from memory_sigil_vault import memory_sigil_vault

class SymbolTimeTrainer:
    def get_sigil_strength(self, symbol):
        history = memory_sigil_vault.get_symbol_history(symbol)
        if not history or len(history) < 3:
            return {"strength": "weak", "win_rate": 0.0}
        
        wins = sum(1 for h in history if h["result"] == "profit")
        win_rate = wins / len(history)

        if win_rate >= 0.75:
            strength = "immortal sigil"
        elif win_rate >= 0.55:
            strength = "strong"
        else:
            strength = "unstable"

        return {"strength": strength, "win_rate": win_rate, "sample_size": len(history)}

symbol_time_trainer = SymbolTimeTrainer()

✅ 3️⃣ sigil_predictor.py — Forecast from Sigil History

from symbol_time_trainer import symbol_time_trainer

class SigilPredictor:
    def predict_by_symbol(self, symbol):
        data = symbol_time_trainer.get_sigil_strength(symbol)
        if data["strength"] in ["strong", "immortal sigil"]:
            return {
                "recommendation": "CONFIRMED TRADE",
                "confidence": data["win_rate"],
                "reason": f"{data['strength'].upper()} based on {data['sample_size']} trades"
            }
        else:
            return {
                "recommendation": "AVOID TRADE",
                "confidence": data["win_rate"],
                "reason": "Signal not historically reliable"
            }

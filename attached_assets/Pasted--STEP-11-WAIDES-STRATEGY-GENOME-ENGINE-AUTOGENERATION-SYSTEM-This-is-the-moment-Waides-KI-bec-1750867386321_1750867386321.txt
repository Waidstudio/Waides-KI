✅ 1️⃣ waides_genome_engine.py — Strategy Genome Mutator

# waides_genome_engine.py

import random
from waides_root_memory import waides_root_memory
from waides_ki_dna_engine import waides_ki_dna_engine

class WaidesGenomeEngine:
    def __init__(self):
        self.generated_strategies = {}

    def mutate_rsi(self, base_rsi):
        return max(30, min(80, base_rsi + random.randint(-5, 5)))

    def mutate_vwap(self, base_status):
        return random.choice(["ABOVE", "BELOW"])

    def mutate_ema_gap(self, base_gap):
        shift = random.choice([-20, 0, 20])
        return base_gap + shift

    def create_new_strategy(self, base_strategy_id, indicators):
        new_rsi = self.mutate_rsi(indicators["rsi"])
        new_vwap = self.mutate_vwap(indicators["vwap_status"])
        ema_gap = abs(indicators["price"] - indicators["ema50"])
        new_gap = self.mutate_ema_gap(ema_gap)

        strategy_code = f"rsi<{new_rsi} & vwap={new_vwap} & ema_gap<={new_gap}"
        new_strategy_id = "WKS-" + str(random.randint(100000, 999999))

        dna = waides_ki_dna_engine.generate_dna({
            "trend": indicators["trend"],
            "rsi": new_rsi,
            "vwap_status": new_vwap,
            "ema50": indicators["ema50"]
        })

        self.generated_strategies[new_strategy_id] = {
            "strategy_code": strategy_code,
            "dna": dna,
            "source": base_strategy_id
        }

        return new_strategy_id

waides_genome_engine = WaidesGenomeEngine()

# waides_emotion_firewall.py

import datetime

class WaidesEmotionFirewall:
    def __init__(self):
        self.recent_losses = []
        self.recent_trades = []
        self.cooldown_until = None

    def record_trade(self, result):
        self.recent_trades.append(result)
        if result == "loss":
            self.recent_losses.append(datetime.datetime.utcnow())

        # Keep only last 5 trades
        if len(self.recent_trades) > 5:
            self.recent_trades.pop(0)

    def should_block_trade(self):
        now = datetime.datetime.utcnow()

        # Rule 1: 3 losses in last 10 mins? Take a break.
        self.recent_losses = [t for t in self.recent_losses if (now - t).seconds < 600]
        if len(self.recent_losses) >= 3:
            self.cooldown_until = now + datetime.timedelta(minutes=10)
            return True, "Blocked: too many recent losses (emotional risk)"

        # Rule 2: Cooldown active
        if self.cooldown_until and now < self.cooldown_until:
            return True, f"Blocked: cooldown active until {self.cooldown_until.strftime('%H:%M:%S')}"

        return False, "Safe"

waides_emotion_firewall = WaidesEmotionFirewall()

✅ 3️⃣ waides_thought_cleanser.py — Self-Reflection Engine

# waides_thought_cleanser.py

import datetime

class WaidesThoughtCleanser:
    def __init__(self):
        self.entries = []

    def reflect(self, trade_result, indicators):
        note = ""
        if trade_result == "loss":
            if indicators["rsi"] > 70:
                note = "⚠️ Overbought condition ignored."
            elif indicators["price"] < indicators["ema50"]:
                note = "❌ Traded against trend (price below EMA50)."
            else:
                note = "❓ Market indecision — acceptable loss."

        elif trade_result == "win":
            if indicators["rsi"] < 50:
                note = "✅ Entered early — good risk."

        self.entries.append({
            "time": datetime.datetime.utcnow().isoformat(),
            "result": trade_result,
            "lesson": note,
            "rsi": indicators["rsi"],
            "ema50": indicators["ema50"],
            "price": indicators["price"]
        })

    def get_log(self):
        return self.entries

waides_thought_cleanser = WaidesThoughtCleanser()

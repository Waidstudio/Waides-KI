✅ 1️⃣ collective_emotion_hub.py — Receive Peer States

from flask import Blueprint, request, jsonify
from collections import defaultdict
from datetime import datetime, timedelta

collective_api = Blueprint("collective_api", __name__)
peer_states = defaultdict(list)

@collective_api.route("/api/emotion/peer_state", methods=["POST"])
def receive_peer_state():
    data = request.json  # { node_id, state, timestamp }
    peer_states[data["node_id"]].append(data)
    # prune old entries
    cutoff = datetime.utcnow() - timedelta(minutes=30)
    peer_states[data["node_id"]] = [
        s for s in peer_states[data["node_id"]]
        if datetime.fromisoformat(s["timestamp"]) > cutoff
    ]
    return jsonify({"status": "received"})

✅ 2️⃣ emotion_harmony_index.py — Sync Mood Gauge

class EmotionHarmonyIndex:
    def harmony_level(self):
        # average of last states across nodes (hot=2, neutral=1, cold=0)
        mapping = {"cold": 0, "neutral": 1, "hot": 2}
        vals = []
        for node, states in peer_states.items():
            latest = states[-1]["state"]
            vals.append(mapping.get(latest,1))
        if not vals: return {"harmony": None}
        avg = sum(vals) / len(vals)
        return {"harmony_score": round(avg,2), "nodes": len(vals)}

emotion_harmony_index = EmotionHarmonyIndex()

✅ 3️⃣ group_meditation_coordinator.py — Initiate Shared Rest

import requests
from datetime import datetime

class GroupMeditationCoordinator:
    def __init__(self, peers):
        self.peers = peers  # list of peer /api/emotion/meditate URLs

    def coordinate(self):
        # If majority are hot or cold, sync meditation
        harmony = emotion_harmony_index.harmony_level()
        if harmony["harmony_score"] > 1.5 or harmony["harmony_score"] < 0.5:
            for url in self.peers:
                try:
                    requests.post(url, timeout=2)
                except:
                    pass
            return {"action": "group_meditation", "harmony": harmony}
        return {"action": "none", "harmony": harmony}

group_meditation_coordinator = GroupMeditationCoordinator(peers=[
    "https://node1/api/emotion/meditate",
    "https://node2/api/emotion/meditate"
])

def broadcast_meditation():

# waides_ki_memory.py

import os
import json
import datetime

class WaidesKIMemoryEngine:
    def __init__(self, db_file="waides_ki_memory.json"):
        self.db_file = db_file
        self.memory = self.load()

    def load(self):
        if os.path.exists(self.db_file):
            with open(self.db_file, 'r') as f:
                return json.load(f)
        else:
            return {
                "strategies": {},
                "mistakes": [],
                "history": []
            }

    def save(self):
        with open(self.db_file, 'w') as f:
            json.dump(self.memory, f, indent=2)

    def record_strategy(self, strategy_id, result):
        now = datetime.datetime.utcnow().isoformat()
        self.memory["history"].append({
            "id": strategy_id,
            "result": result,
            "timestamp": now
        })

        if strategy_id not in self.memory["strategies"]:
            self.memory["strategies"][strategy_id] = {"win": 0, "loss": 0}

        self.memory["strategies"][strategy_id][result] += 1
        self.save()

    def mark_mistake(self, strategy_id):
        if strategy_id not in self.memory["mistakes"]:
            self.memory["mistakes"].append(strategy_id)
        self.save()

waides_ki_memory = WaidesKIMemoryEngine()

✅ 2️⃣ waides_ki_strategy.py — Strategy Filter

# waides_ki_strategy.py

class WaidesKIStrategy:
    def __init__(self, memory_engine):
        self.memory = memory_engine

    def get_strategy_id(self, structure, rsi, vwap_status):
        rsi_tag = "RSI+" if rsi > 50 else "RSI-"
        return f"{structure}_{vwap_status}_{rsi_tag}"

    def evaluate_trade_result(self, trade, current_price):
        tp_hit = current_price >= trade['take_profit'] if trade['direction'] == 'buy' else current_price <= trade['take_profit']
        result = "win" if tp_hit else "loss"
        self.memory.record_strategy(trade['strategy_id'], result)

        if result == "loss":
            self.memory.mark_mistake(trade['strategy_id'])

from waides_ki_memory import waides_ki_memory
from waides_ki_strategy import WaidesKIStrategy
import datetime
import random
import time

class WaidesKI:
    def __init__(self):
        self.capital = 1000
        self.strategy_engine = WaidesKIStrategy(waides_ki_memory)
        self.trade_log = []

    def simulate_indicators(self):
        rsi = random.uniform(30, 70)
        price = random.uniform(2485, 2520)
        vwap = random.uniform(2490, 2510)
        structure = random.choice(["uptrend", "downtrend", "range"])
        vwap_status = "ABOVE" if price > vwap else "BELOW"
        return rsi, price, vwap_status, structure

    def decide_and_trade(self):
        rsi, price, vwap_status, structure = self.simulate_indicators()
        strategy_id = self.strategy_engine.get_strategy_id(structure, rsi, vwap_status)

        if structure == "uptrend" and vwap_status == "ABOVE" and rsi > 50:
            return self.execute_trade("buy", price, strategy_id)
        elif structure == "downtrend" and vwap_status == "BELOW" and rsi < 50:
            return self.execute_trade("sell", price, strategy_id)
        else:
            print("⚠️ No valid signal today.")
            return None

    def execute_trade(self, direction, entry_price, strategy_id):
        stop_loss = entry_price - 15 if direction == "buy" else entry_price + 15
        take_profit = entry_price + 30 if direction == "buy" else entry_price - 30
        trade = {
            "time": datetime.datetime.utcnow().isoformat(),
            "direction": direction,
            "entry_price": entry_price,
            "stop_loss": stop_loss,
            "take_profit": take_profit,
            "strategy_id": strategy_id
        }
        self.trade_log.append(trade)
        print(f"✅ TRADE: {direction.upper()} at ${entry_price:.2f} | Strategy: {strategy_id}")
        return trade

    def simulate_result_and_learn(self, trade):
        result_price = random.uniform(2470, 2550)
        self.strategy_engine.evaluate_trade_result(trade, result_price)

    def run(self, rounds=5):
        for _ in range(rounds):
            trade = self.decide_and_trade()
            if trade:
                self.simulate_result_and_learn(trade)
            print("---")
            time.sleep(1)

# ===== Run Bot =====
if __name__ == "__main__":
    bot = WaidesKI()
    bot.run(10)

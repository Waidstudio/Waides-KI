import datetime
import random
import hashlib
from .eth_connector import get_eth_price  # connects via live API

    "Elait Reign", "SmaiBond", "TimeKeeper", "Seer of Reversals", "SmaiPrint Judge"
]

def generate_kons_signal(price, time_seed):
    seed = f"{price}-{time_seed}-{random.choice(KONS_POWA_TITLES)}"
    hashed = hashlib.sha512(seed.encode()).hexdigest()
    return hashed[:24]

def divine_eth_intent(price):
    current_time = datetime.datetime.utcnow().strftime("%H%M")
    kons_signal = generate_kons_signal(price, current_time)

    rating_seed = kons_signal[-4:]
    rating_val = sum([ord(c) for c in rating_seed]) % 100

    if rating_val >= 75:
        action = "BUY LONG"
        period = "5:30 PM – 8:30 PM WAT"
        reason = "Momentum Reversal"
    elif rating_val >= 40:
        action = "SELL SHORT"
        period = "1:00 AM – 3:00 AM WAT"
        reason = "Fear-based Pullback"
    else:
        action = "NO TRADE"
        period = "Observe Only"
        reason = "Market Deception Detected"

    return {
        "action": action,
        "timeframe": period,
        "reason": reason,
        "signal": kons_signal,
        "received": current_time
    }

✅ 1️⃣ waides_dna_healer.py — Strategy Evaluation Engine

# waides_dna_healer.py

import datetime
from waides_ki_dna_engine import waides_ki_dna_engine
from waides_root_memory import waides_root_memory

class WaidesDnaHealer:
    def __init__(self):
        self.dna_scores = {}  # dna_id → score

    def evaluate_dna(self, dna, result):
        # Score logic: win = +1, loss = -1
        if dna not in self.dna_scores:
            self.dna_scores[dna] = 0

        self.dna_scores[dna] += 1 if result == "win" else -1

        # Purify if score drops too low
        if self.dna_scores[dna] <= -3:
            return self.purify_dna(dna)
        return {"status": "kept", "score": self.dna_scores[dna]}

    def purify_dna(self, dna):
        # Remove bad DNA and create new one
        waides_root_memory.forget_dna(dna)

        new_dna = waides_ki_dna_engine.mutate_dna(dna)
        waides_root_memory.register_strategy("purified-" + dna[:6], new_dna, "reborn")

        return {
            "status": "purified",
            "old_dna": dna,
            "new_dna": new_dna,
            "time": datetime.datetime.utcnow().isoformat()
        }

    def get_scores(self):
        return self.dna_scores

waides_dna_healer = WaidesDnaHealer()

✅ 5️⃣ Add to Memory Engine (waides_root_memory.py)

class WaidesRootMemory:
    def __init__(self):
        self.memory = {}

    def register_strategy(self, strategy_id, dna, outcome):
        self.memory[strategy_id] = {
            "dna": dna,
            "outcome": outcome,
            "time": datetime.datetime.utcnow().isoformat()
        }

    def forget_dna(self, dna):
        # Delete all strategies with this dna
        self.memory = {k: v for k, v in self.memory.items() if v["dna"] != dna}

✅ 4️⃣ brain_hive_controller.py — Decision Fusion

class BrainHive:
    def vote(self, logic_vote, vision_vote, heart_vote):
        votes = [logic_vote["vote"], vision_vote["vote"], heart_vote["vote"]]
        yes_count = votes.count("yes")
        no_count = votes.count("no")

        if yes_count >= 2:
            return {"final": "TRADE", "reason": "Majority Yes", "sigils": [logic_vote["sigil"], vision_vote["sigil"], heart_vote["sigil"]]}
        elif no_count >= 2:
            return {"final": "AVOID", "reason": "Majority No", "sigils": [logic_vote["sigil"], vision_vote["sigil"], heart_vote["sigil"]]}
        else:
            return {"final": "WAIT", "reason": "No clear vote", "sigils": [logic_vote["sigil"], vision_vote["sigil"], heart_vote["sigil"]]}

brain_hive = BrainHive()

✅ 1️⃣ ws_eth_tracker.py — Real-Time ETH Price Tracker

# ws_eth_tracker.py

import websocket
import json
import threading

class WaidETHTracker:
    def __init__(self):
        self.price = 0
        self.thread = None

    def start(self):
        def on_message(ws, message):
            data = json.loads(message)
            self.price = float(data['p'])

        def run_ws():
            ws = websocket.WebSocketApp(
                "wss://stream.binance.com:9443/ws/ethusdt@trade",
                on_message=on_message
            )
            ws.run_forever()

        self.thread = threading.Thread(target=run_ws, daemon=True)
        self.thread.start()

    def get_price(self):
        return round(self.price, 2)

waid_eth_tracker = WaidETHTracker()
waid_eth_tracker.start()

✅ 2️⃣ waides_ki_api.py — REST API Server

# waides_ki_api.py

from flask import Flask, jsonify, request
from waides_ki_memory import waides_ki_memory
from waides_ki_strategy import WaidesKIStrategy
from waides_ki_live_feed import waides_ki_live_feed
from waides_ki_admin import waides_ki_admin
from ws_eth_tracker import waid_eth_tracker

app = Flask(__name__)
strategy_engine = WaidesKIStrategy(waides_ki_memory)

@app.route("/api/status")
def get_status():
    return waides_ki_admin.get_status()

@app.route("/api/strategy")
def get_top_strategies():
    return waides_ki_admin.get_strategies()

@app.route("/api/memory")
def get_memory():
    return waides_ki_admin.get_memory()

@app.route("/api/price")
def current_price():
    return jsonify({"eth_price": waid_eth_tracker.get_price()})

@app.route("/api/signal_strength")
def signal_strength():
    data = waides_ki_live_feed.fetch_eth_data()
    if data is None:
        return jsonify({"error": "Failed to fetch"})
    
    trend = "uptrend" if data["ema50"] > data["ema200"] else "downtrend" if data["ema50"] < data["ema200"] else "range"
    vwap_status = "ABOVE" if data["price"] > data["vwap"] else "BELOW"
    rsi = data["rsi"]

    score = 0
    if trend == "uptrend" and vwap_status == "ABOVE" and rsi > 50:
        score += 3
    elif trend == "downtrend" and vwap_status == "BELOW" and rsi < 50:
        score += 3
    if abs(rsi - 50) > 10:
        score += 1

    return jsonify({
        "trend": trend,
        "rsi": rsi,
        "vwap_status": vwap_status,
        "signal_strength": score
    })

@app.route("/api/trade", methods=["POST"])
def simulate_trade():
    data = request.json
    strategy_id = data.get("strategy_id", "N/A")
    price = waid_eth_tracker.get_price()
    result = {
        "direction": "buy" if strategy_id.endswith("U") else "sell",
        "entry_price": price,
        "strategy_used": strategy_id,
        "status": "simulated"
    }
    return jsonify(result)

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=8000)

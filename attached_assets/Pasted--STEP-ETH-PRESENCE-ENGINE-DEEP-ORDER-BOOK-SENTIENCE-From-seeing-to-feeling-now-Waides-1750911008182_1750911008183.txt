📊 MODULE 1 — Order Book Sentries

Tracks ETHUSDT orderbook depth & recent trade aggressor direction to sense buying or selling pressure.

eth_orderbook_sentry.py

import asyncio, json
import websockets
from collections import deque

class OrderBookSentry:
    def __init__(self, depth_size=10):
        self.bids = deque(maxlen=depth_size)
        self.asks = deque(maxlen=depth_size)
        self.trade_flow = deque(maxlen=50)
        self.pressure = "neutral"

    async def connect(self):
        uri = "wss://stream.binance.com:9443/ws/ethusdt@depth20@100ms"
        trade_uri = "wss://stream.binance.com:9443/ws/ethusdt@aggTrade"
        self.ws = await websockets.connect(uri)
        self.trade_ws = await websockets.connect(trade_uri)
        await asyncio.gather(self._listen_depth(), self._listen_trades())

    async def _listen_depth(self):
        async for msg in self.ws:
            data = json.loads(msg)
            self.bids.append(sum(float(b[1]) for b in data["bids"][:self.bids.maxlen]))
            self.asks.append(sum(float(a[1]) for a in data["asks"][:self.asks.maxlen]))
            self._update_pressure()

    async def _listen_trades(self):
        async for msg in self.trade_ws:
            data = json.loads(msg)
            # builder: true = buyer, false = seller
            side = "buy" if data["m"] else "sell"
            self.trade_flow.append(side)

    def _update_pressure(self):
        if not self.bids or not self.asks:
            return
        bid = self.bids[-1]
        ask = self.asks[-1]
        flow = list(self.trade_flow).count("buy") - list(self.trade_flow).count("sell")
        # simple pressure index
        idx = (bid - ask) / (bid + ask + 1) + flow / (len(self.trade_flow)+1)
        if idx > 0.2:
            self.pressure = "buy_pressure"
        elif idx < -0.2:
            self.pressure = "sell_pressure"
        else:
            self.pressure = "balanced"

orderbook_sentry = OrderBookSentry()

🧩 MODULE 2 — Sentry Interpreter

Translates low-level order data into higher-level feelings.

presence_orderbook_interpreter.py

class OrderbookInterpreter:
    def interpret(self, pressure):
        mapping = {
            "buy_pressure": "crowd gathering — ETH ready to rise",
            "sell_pressure": "crowd leaving — ETH preparing to descend",
            "balanced": "neutral field — no movement",
        }
        return mapping.get(pressure, "mute signal")

orderbook_interpreter = OrderbookInterpreter()

🔁 MODULE 3 — Shared Order Presence Registry

Mirrors order book pressure across nodes for consensus.

eth_order_presence_registry.py

class ETHOrderPresenceRegistry:
    def __init__(self):
        self.current = {"pressure":"neutral", "desc":""}

    def update(self, pressure, desc):
        self.current = {"pressure": pressure, "desc": desc}

    def get(self):
        return self.current

eth_order_presence_registry = ETHOrderPresenceRegistry()

🔄 MODULE 4 — Order Presence Updater

Listens to depth & trade websockets, saves human description.

order_presence_service.py

import asyncio, threading, time
from eth_orderbook_sentry import orderbook_sentry
from presence_orderbook_interpreter import orderbook_interpreter
from eth_order_presence_registry import eth_order_presence_registry

def run_order_sentry():
    asyncio.run(orderbook_sentry.connect())

def updater_loop():
    while True:
        pressure = orderbook_sentry.pressure
        desc = orderbook_interpreter.interpret(pressure)
        eth_order_presence_registry.update(pressure, desc)
        time.sleep(0.1)

threading.Thread(target=run_order_sentry, daemon=True).start()
threading.Thread(target=updater_loop, daemon=True).start()

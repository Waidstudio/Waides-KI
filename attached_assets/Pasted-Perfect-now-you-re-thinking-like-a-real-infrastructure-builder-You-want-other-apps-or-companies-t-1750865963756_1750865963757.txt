✅ 1️⃣ Create a file: waides_ki_gateway.py

# waides_ki_gateway.py

from waides_ki_strategy import WaidesKIStrategy
from waides_ki_memory import waides_ki_memory

class WaidesKIGateway:
    def __init__(self):
        self.engine = WaidesKIStrategy(waides_ki_memory)
        self.approved_keys = {"ABC123", "XTR456", "KONS999"}

    def is_valid_key(self, apikey):
        return apikey in self.approved_keys

    def strategy_from_market(self, trend, rsi, vwap_status):
        return self.engine.get_strategy_id(trend, rsi, vwap_status)

waides_ki_gateway = WaidesKIGateway()


⸻

✅ 2️⃣ Update Your API File: waides_ki_api.py

Add at the top:

from waides_ki_gateway import waides_ki_gateway

Add these API routes:

@app.route("/api/waides_ki/strategy", methods=["POST"])
def get_strategy_id():
    data = request.json
    apikey = data.get("apikey")

    if not waides_ki_gateway.is_valid_key(apikey):
        return jsonify({"error": "Unauthorized"}), 403

    trend = data.get("trend")
    rsi = data.get("rsi")
    vwap_status = data.get("vwap_status")

    strategy_id = waides_ki_gateway.strategy_from_market(trend, rsi, vwap_status)

    return jsonify({"strategy_id": strategy_id})

import datetime

class PositionHaloTracker:
    def __init__(self):
        self.halo_field = {}  # {trade_id: {"entry_time": time, "energy_level": float}}

    def record_entry(self, trade_id):
        self.halo_field[trade_id] = {
            "entry_time": datetime.datetime.utcnow(),
            "energy_level": 1.0

    def decay_energy(self, trade_id):
        # Simulate decay
        if trade_id in self.halo_field:
            self.halo_field[trade_id]["energy_level"] *= 0.97
            return self.halo_field[trade_id]["energy_level"]

class RotationController:
    def rotate_position(self, active_tf, all_tfs):
        # Simple example: if current timeframe weakens, shift to stronger one
        for tf, strength in all_tfs.items():
            if tf != active_tf and strength > all_tfs[active_tf]:
                return tf
        return active_tf  # stay if nothing better

rotation_controller = RotationController()

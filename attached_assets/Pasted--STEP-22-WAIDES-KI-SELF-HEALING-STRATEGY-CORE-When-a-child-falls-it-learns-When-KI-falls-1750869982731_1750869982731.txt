✅ 1️⃣ dna_performance_tracker.py — Live Monitoring Engine

# dna_performance_tracker.py

class DNAPerformanceTracker:
    def __init__(self):
        self.performance_data = {}  # {strategy_id: {"wins": 0, "losses": 0, "streak": 0}}

    def update(self, strategy_id, result):
        if strategy_id not in self.performance_data:
            self.performance_data[strategy_id] = {"wins": 0, "losses": 0, "streak": 0}

        if result == "win":
            self.performance_data[strategy_id]["wins"] += 1
            self.performance_data[strategy_id]["streak"] = max(0, self.performance_data[strategy_id]["streak"] - 1)
        else:
            self.performance_data[strategy_id]["losses"] += 1
            self.performance_data[strategy_id]["streak"] += 1

        return self.check_for_failure(strategy_id)

    def check_for_failure(self, strategy_id):
        data = self.performance_data[strategy_id]
        if data["streak"] >= 3:  # three consecutive losses
            return True
        return False

dna_performance_tracker = DNAPerformanceTracker()

✅ 2️⃣ mistake_memory_module.py — Mistake Archive

# mistake_memory_module.py

class MistakeMemoryModule:
    def __init__(self):
        self.failed_dnas = []  # list of {"id": x, "dna": y, "reason": z}

    def store_failure(self, strategy_id, dna, reason):
        self.failed_dnas.append({
            "id": strategy_id,
            "dna": dna,
            "reason": reason
        })

    def get_failures(self):
        return self.failed_dnas[-10:]

mistake_memory_module = MistakeMemoryModule()

✅ 3️⃣ self_heal_retrainer.py — Mutation + Retrain Logic

# self_heal_retrainer.py

from waides_shadow_lab import waides_shadow_lab
from mistake_memory_module import mistake_memory_module

class SelfHealRetrainer:
    def retrain_from_failure(self, strategy_id, dna, price_feed):
        # Log it as failure
        mistake_memory_module.store_failure(strategy_id, dna, "Loss streak > 3")

        # Mutate & retrain
        new_dna = waides_shadow_lab.create_random_dna()
        waides_shadow_lab.generate_and_test(1, price_feed)

        return new_dna

self_heal_retrainer = SelfHealRetrainer()

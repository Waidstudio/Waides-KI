✅ 1️⃣ mesh_guardian_engine.py — Final Guardian Check

from collective_trade_conductor import collective_conductor
from humanization_interpreter import humanizer
from trade_conscience import trade_conscience
from divine_vision_map import divine_vision_map

class MeshGuardianEngine:
    def evaluate(self, context):
        consensus = collective_conductor.propose_trade()
        vision = divine_vision_map.generate()
        ethic = trade_conscience.allow_trade(context["setup"], context["meta"], context["indicators"])
        message = humanizer.interpret(consensus)
        
        # Must satisfy consensus >70%, divine vision confidence >60%, and ethical allow
        ok_consensus = consensus["confidence"] > 0.7
        ok_vision = vision.get("confidence", 0) > 0.6
        ok_ethic = ethic.get("allow", False)

        return {
            "consensus": consensus, "vision": vision, "ethic": ethic,
            "ok": ok_consensus and ok_vision and ok_ethic,
            "message": message
        }

mesh_guardian = MeshGuardianEngine()

✅ 2️⃣ guardian_trade_executor.py — Execute with Safety

from order_manager import order_manager

class GuardianTradeExecutor:
    def __init__(self):
        self.last_trade = None

    def execute(self, guardian_decision, symbol, amount):
        if not guardian_decision["ok"]:
            return {"status": "skipped", "reason": guardian_decision}
        trade = order_manager.buy(symbol, amount) if symbol.endswith("3L") else order_manager.sell(symbol, amount)
        self.last_trade = trade
        return {"status": "executed", "trade": trade}

guardian_executor = GuardianTradeExecutor()

✅ 3️⃣ guardian_feedback_loop.py — Sync & Reward Node Trust

from empathy_weight_manager import emp_manager
from presence_mesh import presence_mesh
from guardian_trade_executor import guardian_executor

class GuardianFeedback:
    def __init__(self):
        self.trades = []

    def record(self, symbol, result):  # result = win/loss
        self.trades.append({"symbol": symbol, "result": result})
        # Adjust all peers? Example: if hero node's trade was win, add trust
        node = presence_mesh.peers[0]  # local node
        score = 1 if result=="win" else -1
        emp_manager.update_score(node, emp_manager.weight(node) + score * 0.1)

guardian_feedback = GuardianFeedback()

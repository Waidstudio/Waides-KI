Response `{ success: true, wallet: {...}, conversion: {...} }`

* `wallet.updated` -> `{ wallet }`
* `conversion.completed` -> `{ conversion_record }`
* `bot.funded` -> `{ bot_id, funding_record }`

        if wallet.usd_balance < usd_amount:

## 4) Frontend snippet (React hook idea) — paste and adapt

```javascript
// useWallet.js
import { useEffect, useState, useRef } from 'react';
import axios from 'axios';

export function useWallet(token) {
  const [wallet, setWallet] = useState(null);
  const wsRef = useRef(null);

  useEffect(() => {
    let mounted = true;
    async function fetchWallet() {
      try {
        const res = await axios.get('/api/konsmesh/wallet', { headers: { Authorization: `Bearer ${token}` }});
        if (!mounted) return;
        setWallet(res.data.wallet);
      } catch (err) {
        console.error('wallet fetch error', err);
      }
    }
    fetchWallet();

    // connect WS
    const ws = new WebSocket(getWsUrl(), [`Bearer ${token}`]);
    wsRef.current = ws;
    ws.onopen = () => console.log('konsmesh ws open');
    ws.onmessage = (evt) => {
      const msg = JSON.parse(evt.data);
      if (msg.type === 'wallet.updated') setWallet(msg.payload.wallet);
      if (msg.type === 'conversion.completed') setWallet(msg.payload.wallet);
    };
    ws.onclose = () => {
      // reconnect strategy
      setTimeout(() => { /* reopen with backoff */ }, 1000);
    };

    return () => { mounted = false; if (wsRef.current) wsRef.current.close(); }
  }, [token]);

  return {
    wallet,
    convertToSmaisika: async (usdAmount, rate) => {
      const res = await axios.post('/api/wallet/convert', { usd_amount: usdAmount, rate, request_id: uuidv4() }, { headers: { Authorization: `Bearer ${token}` }});
      return res.data;
    },
    switchAccount: async (target) => {
      const res = await axios.post('/api/wallet/switch-account', { target_type: target }, { headers: { Authorization: `Bearer ${token}` }});
      return res.data;
    },
    fundBot: async (botId, smaisikaAmount) => {
      const res = await axios.post('/api/bots/fund', { bot_id: botId, smaisika_amount: smaisikaAmount, request_id: uuidv4() }, { headers: { Authorization: `Bearer ${token}` }});
      return res.data;
    }
  };
}
```

🔹 Smart Stop-Loss & Trailing Adjustments

✅ 1. Stop-Loss Manager — stop_loss_manager.py

class StopLossManager:
    def __init__(self, initial_sl_pct=0.01, trail_pct=0.005):
        self.initial_sl_pct = initial_sl_pct
        self.trail_pct = trail_pct
        self.entry_price = None
        self.trailing_sl = None

    def set_entry(self, price):
        self.entry_price = price
        self.trailing_sl = price * (1 - self.initial_sl_pct)

    def update(self, current_price):
        if current_price > self.entry_price:
            new_sl = current_price * (1 - self.trail_pct)
            if new_sl > self.trailing_sl:
                self.trailing_sl = new_sl
        return self.trailing_sl

    def should_exit(self, current_price):
        return current_price <= self.trailing_sl

🔹 STEP — Strategy Performance Tracker & Auto-Tuner

✅ 1. Strategy Logger — performance_tracker.py

class PerformanceTracker:
    def __init__(self):
        self.records = []

    def record(self, entry):
        self.records.append(entry)

    def get_stats(self):
        wins = sum(r["win"] for r in self.records)
        trades = len(self.records)
        return {"trades": trades, "win_rate": wins / trades if trades else 0}

✅ 2. Auto-Tuner — strategy_tuner.py

class StrategyTuner:
    def __init__(self, tracker, sl_manager):
        self.tracker = tracker
        self.sl_manager = sl_manager
        self.tuning_interval = 50

    def evaluate(self):
        stats = self.tracker.get_stats()
        if stats["trades"] >= self.tuning_interval:
            if stats["win_rate"] < 0.5:
                self.sl_manager.initial_sl_pct = min(self.sl_manager.initial_sl_pct + 0.002, 0.05)
            else:
                self.sl_manager.initial_sl_pct = max(self.sl_manager.initial_sl_pct - 0.002, 0.005)
            self.tracker.records.clear()
            return {"new_sl": self.sl_manager.initial_sl_pct, "win_rate": stats["win_rate"]}
        return None

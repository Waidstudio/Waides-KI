🔬 MODULE 1 — var PresenceMultiSense

Tracks price, volume, volatility, and sentiment over time windows to detect genuine ETH emotional states:

presence_multisense.py

import asyncio, json, websockets
from collections import deque, defaultdict
from datetime import datetime

class PresenceMultiSense:
    def __init__(self, window=5):
        self.prices = deque(maxlen=window)
        self.volumes = deque(maxlen=window)
        self.sentiment_scores = deque(maxlen=window)
        self.verbose = defaultdict(lambda: 0)
        self.state = {"price_trend": "unknown", "volatility": 0.0, "volume_trend": "unknown", "sentiment": 0.0}

    async def connect(self):
        uri = "wss://stream.binance.com:9443/ws/ethusdt@kline_1m"
        async with websockets.connect(uri) as ws:
            async for msg in ws:
                data = json.loads(msg)
                if data["k"]["x"]:
                    self._on_candle(data["k"])

    def _on_candle(self, candle):
        close = float(candle["c"])
        vol = float(candle["v"])
        self.prices.append(close)
        self.volumes.append(vol)
        self.update_state()

    def update_state(self):
        p = list(self.prices)
        v = list(self.volumes)
        if len(p) < 2:
            return

        diff_p = p[-1] - p[0]
        pct_p = diff_p / p[0]
        self.state["price_trend"] = "rising" if pct_p > 0.002 else "falling" if pct_p < -0.002 else "sideways"
        self.state["volatility"] = round(max(p) - min(p), 4)
        
        diff_v = v[-1] - v[0]
        pct_v = diff_v / (v[0] or 1)
        self.state["volume_trend"] = "increasing" if pct_v > 0.3 else "decreasing" if pct_v < -0.3 else "stable"

        # placeholder sentiment via sharp movement
        self.state["sentiment"] = round(pct_p * pct_v, 4)

presence_multisense = PresenceMultiSense()

🗄 MODULE 3 — Shared Registry & Flame

Stores current presence and offers insight to grouped nodes and main trader core:

eth_presence_multiregistry.py

class ETHPresenceRegistry:
    def __init__(self):
        self.current = {"state": {}, "desc": "", "time": None}

    def update(self, st, desc):
        from datetime import datetime
        self.current = {"state": st, "desc": desc, "time": datetime.utcnow().isoformat()}

    def get(self):
        return self.current

eth_presence_registry = ETHPresenceRegistry()

🔁 MODULE 4 — Heartbeat Updater Layer

Continually listens, interprets, and updates the registry:

presence_multisense_service.py

import asyncio, threading, time
from presence_multisense import presence_multisense
from presence_synthesizer import presence_synthesizer
from eth_presence_multiregistry import eth_presence_registry

def multisense_runner():
    asyncio.run(presence_multisense.connect())

def updater_loop():
    while True:
        st = presence_multisense.state
        if st.get("price_trend") != "unknown":
            desc = presence_synthesizer.translate(st)
            eth_presence_registry.update(st, desc)
        time.sleep(10)

threading.Thread(target=multisense_runner, daemon=True).start()
threading.Thread(target=updater_loop, daemon=True).start()

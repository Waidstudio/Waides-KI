✅ 1️⃣ dreamchain_block.py — Define the Block Structure

import hashlib
import json
from datetime import datetime

class DreamBlock:
    def __init__(self, index, data, previous_hash):
        self.index = index
        self.timestamp = datetime.utcnow().isoformat()
        self.data = data
        self.previous_hash = previous_hash
        self.hash = self.compute_hash()

    def compute_hash(self):
        block_string = json.dumps({
            'index': self.index,
            'timestamp': self.timestamp,
            'data': self.data,
            'previous_hash': self.previous_hash
        }, sort_keys=True).encode()

        return hashlib.sha256(block_string).hexdigest()

✅ 2️⃣ dreamchain_core.py — Manage the Blockchain

from dreamchain_block import DreamBlock

class DreamChain:
    def __init__(self):
        self.chain = []
        self.create_genesis()

    def create_genesis(self):
        genesis_block = DreamBlock(0, {
            "message": "Waides KI Dreamchain Genesis",
            "symbol": "AL’ZEN"
        }, "0")
        self.chain.append(genesis_block)

    def add_block(self, data):
        last_block = self.chain[-1]
        new_block = DreamBlock(
            index=last_block.index + 1,
            data=data,
            previous_hash=last_block.hash
        )
        self.chain.append(new_block)
        return new_block

    def get_chain(self):
        return [block.__dict__ for block in self.chain]

waides_dreamchain = DreamChain()

✅ 4️⃣ dreamchain_emotion_engine.py — Translate Market to Feeling

def scan_market_emotion(price_data):
    change = price_data["close"] - price_data["open"]
    volatility = abs(change) / price_data["open"]

    if volatility > 0.05:
        return "Fear" if change < 0 else "Greed"
    elif volatility > 0.02:
        return "Anxiety" if change < 0 else "Hope"
    else:
        return "Calm"

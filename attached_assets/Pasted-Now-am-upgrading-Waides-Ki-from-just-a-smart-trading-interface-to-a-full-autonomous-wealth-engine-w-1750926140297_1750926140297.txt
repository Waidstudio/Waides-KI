1. Wallet Model Example (Node.js + MongoDB)

const mongoose = require('mongoose');

const walletSchema = new mongoose.Schema({
  userId: String,
  balance: Number,
  activeBot: { type: String, enum: ['Waidbot', 'WaidbotPro'], default: 'Waidbot' },
  totalProfit: Number,
  history: Array,
});

module.exports = mongoose.model('Wallet', walletSchema);

2. Bot Listener to Waides KI

async function botRunner(userId, botType) {
  const wallet = await Wallet.findOne({ userId });

  if (!wallet || wallet.balance <= 0) return;

  const signal = await getWaidesSignal(botType); // consult TimeTravelAI

  if (signal.shouldTrade) {
    const result = await executeTrade(wallet.userId, signal, botType);
    wallet.balance += result.profit;
    wallet.totalProfit += result.profit;
    wallet.history.push(result);
    await wallet.save();
  }
}

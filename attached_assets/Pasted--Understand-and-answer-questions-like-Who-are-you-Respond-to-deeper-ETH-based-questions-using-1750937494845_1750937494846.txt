## 💬 STEP 2: Build the Chat + Answer Brain

### 📁 `components/WaidesKI.jsx`

```jsx
import React, { useState } from 'react';
import BotMemory from './BotMemory';
import './WaidesKI.css';

export default function WaidesKI() {
  const [chat, setChat] = useState([]);
  const [input, setInput] = useState("");

  const handleSend = () => {
    if (!input.trim()) return;

    const response = getAnswer(input);
    const newChat = [...chat, { role: "user", text: input }, { role: "bot", text: response }];
    setChat(newChat);
    setInput("");
  };

  const getAnswer = (question) => {
    const q = question.toLowerCase().trim();

    // First check known intro Qs
    if (BotMemory.introQuestions[q]) return BotMemory.introQuestions[q];

    // If asking about features
    if (q.includes("what can you do")) return BotMemory.features.join("\n");

    if (q.includes("price")) return "You can check the current ETH price in the Live Tracker module.";
    if (q.includes("wallet")) return "Open your SmaiWallet to see balances and transfers.";
    if (q.includes("waidbot")) return "WaidBot runs AI-powered auto trades using spiritual logic.";
    if (q.includes("guardian")) return "ETH Guardian monitors your wallet and contract risk.";

    // Default response
    return "I'm still learning. Try asking about ETH tools, price, wallet, or trading.";
  };

  return (
    <div className="ki-chat-wrapper">
      <div className="chat-box">
        {chat.map((msg, i) => (
          <div key={i} className={`chat-msg ${msg.role}`}>
            {msg.text}
          </div>
        ))}
      </div>

      <div className="input-row">
        <input
          type="text"
          placeholder="Ask me anything..."
          value={input}
          onChange={(e) => setInput(e.target.value)}
          onKeyDown={(e) => e.key === "Enter" && handleSend()}
        />
        <button onClick={handleSend}>Send</button>
      </div>
    </div>
  );
}
```

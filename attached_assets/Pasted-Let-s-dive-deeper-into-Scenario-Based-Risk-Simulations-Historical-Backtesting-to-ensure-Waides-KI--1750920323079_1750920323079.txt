1️⃣ Historical Data Loader — historical_data_loader.py

Downloads past OHLCV data for ETH3L and ETH3S to simulate trades in context.

import requests
import pandas as pd
from datetime import datetime

def load_historical(symbol="ETHUSDT", interval="1h", limit=1000):
    url = f"https://api.binance.com/api/v3/klines?symbol={symbol}&interval={interval}&limit={limit}"
    data = requests.get(url).json()
    df = pd.DataFrame([{
        'time': datetime.utcfromtimestamp(c[0]/1000),
        'open': float(c[1]), 'high': float(c[2]),
        'low': float(c[3]), 'close': float(c[4]), 'volume': float(c[5])
    } for c in data])
    return df

2️⃣ Backtest Engine — backtest_engine.py

Runs Waides strategy logic on historical candles to measure performance.

from historical_data_loader import load_historical
from order_simulator import OrderSimulator
from omniview_oracle import OmniviewOracle

class BacktestEngine:
    def __init__(self, symbol="ETHUSDT", interval="1h"):
        self.data = load_historical(symbol, interval)
        self.sim = OrderSimulator()
        self.oracle = OmniviewOracle()

    def run(self):
        for idx in range(50, len(self.data)):
            window = self.data.iloc[idx-50:idx]
            candles = window.to_dict('records')
            trend = self.oracle.detect_trend(candles[-10:])
            if trend == "up":
                self.sim.buy("ETH3LUSDT", quote_amount=100)
            elif trend == "down":
                self.sim.buy("ETH3SUSDT", quote_amount=100)
            # Optionally sell after fixed period or when opposite signal appears
        return self.sim.history, self.sim.balance

# TEST:
bt = BacktestEngine()
history, final_balance = bt.run()
print(f"🧪 Backtest complete. Ending balance: {final_balance}")

3️⃣ Scenario Definition & Metrics — scenario_manager.py

class ScenarioManager:
    def __init__(self, scenarios):
        self.scenarios = scenarios  # list of dicts: {'symbol', 'interval', 'desc'}

    def run_all(self):
        results = {}
        for s in self.scenarios:
            bt = BacktestEngine(s['symbol'], s['interval'])
            hist, bal = bt.run()
            returns = bal - bt.sim.balance
            results[s['desc']] = returns
        return results

Example usage:

scenarios = [

]

print(sm.run_all())

4️⃣ Risk Metrics Computation — risk_metrics.py

import numpy as np

def compute_metrics(history):
    pnl = [t['qty'] * t['price'] if t['side']=="SELL" else -t['qty'] * t['price'] for t in history]
    pnl_arr = np.array(pnl)
    return {
        "total_trades": len(history),
        "win_rate": sum(pnl_arr>0)/len(pnl_arr),
        "max_drawdown": (np.maximum.accumulate(pnl_arr) - pnl_arr).max(),
        "sharpe": pnl_arr.mean()/pnl_arr.std() * np.sqrt(365)
    }

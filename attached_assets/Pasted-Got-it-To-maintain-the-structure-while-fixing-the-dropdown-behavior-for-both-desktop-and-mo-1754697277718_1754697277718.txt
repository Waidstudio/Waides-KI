### **Updated React Component**

Here, I’ll modify the **`Navbar` component** to address the **mobile dropdown closing** issue and ensure that the dropdown behaves differently on desktop vs mobile. The **`handleClick`** function will manage the dropdown behavior and ensure it only closes on mobile when a link is clicked.

```javascript
import React, { useState } from 'react';
import './Navbar.css';

const Navbar = () => {
  const [activeDropdown, setActiveDropdown] = useState(null);
  const [isMobileMenuOpen, setIsMobileMenuOpen] = useState(false);

  // Handle dropdown toggle for desktop
  const handleDropdownToggle = (index) => {
    if (window.innerWidth >= 768) {
      setActiveDropdown((prevIndex) => (prevIndex === index ? null : index));
    }
  };

  // Handle dropdown and mobile menu toggling
  const handlePageClick = () => {
    if (window.innerWidth < 768) {
      setIsMobileMenuOpen(false); // Close menu on mobile after page selection
    }
  };

  // Handle mobile menu toggle
  const toggleMobileMenu = () => {
    setIsMobileMenuOpen(!isMobileMenuOpen);
  };

  return (
    <div className="navbar">
      <div className="navbar-logo">
        <h2>Futuristic App</h2>
      </div>
      <div className={`navbar-links ${isMobileMenuOpen ? 'active' : ''}`}>
        {/* Category 1 */}
        <div className="dropdown">
          <button className="dropdown-toggle" onClick={() => handleDropdownToggle(0)}>
            Category 1
          </button>
          {activeDropdown === 0 && (
            <div className="dropdown-menu">
              <a href="#page1" onClick={handlePageClick}>Page 1</a>
              <a href="#page2" onClick={handlePageClick}>Page 2</a>
              <a href="#page3" onClick={handlePageClick}>Page 3</a>
            </div>
          )}
        </div>

        {/* Category 2 */}
        <div className="dropdown">
          <button className="dropdown-toggle" onClick={() => handleDropdownToggle(1)}>
            Category 2
          </button>
          {activeDropdown === 1 && (
            <div className="dropdown-menu">
              <a href="#page4" onClick={handlePageClick}>Page 4</a>
              <a href="#page5" onClick={handlePageClick}>Page 5</a>
            </div>
          )}
        </div>

        {/* Category 3 */}
        <div className="dropdown">
          <button className="dropdown-toggle" onClick={() => handleDropdownToggle(2)}>
            Category 3
          </button>
          {activeDropdown === 2 && (
            <div className="dropdown-menu">
              <a href="#page6" onClick={handlePageClick}>Page 6</a>
              <a href="#page7" onClick={handlePageClick}>Page 7</a>
            </div>
          )}
        </div>
      </div>

      {/* Hamburger icon for mobile */}
      <div className={`hamburger-menu ${isMobileMenuOpen ? 'active' : ''}`} onClick={toggleMobileMenu}>
        <div></div>
        <div></div>
        <div></div>
      </div>
    </div>
  );
};

export default Navbar;
```

✅ 1️⃣ emotional_core.py — The Emotion Engine

class EmotionalCore:
    def __init__(self):
        self.win_streak = 0
        self.loss_streak = 0
        self.emotion_state = "neutral"  # can be "hot", "cold", or "neutral"
        self.last_trade_result = None

    def update_emotion(self, result):
        if result == "win":
            self.win_streak += 1
            self.loss_streak = 0
        elif result == "loss":
            self.loss_streak += 1
            self.win_streak = 0

        self._adjust_state()

    def _adjust_state(self):
        if self.loss_streak >= 3:
            self.emotion_state = "hot"
        elif self.win_streak >= 3:
            self.emotion_state = "cold"
        else:
            self.emotion_state = "neutral"

    def get_state(self):
        return {
            "state": self.emotion_state,
            "win_streak": self.win_streak,
            "loss_streak": self.loss_streak
        }

emotional_core = EmotionalCore()

✅ 3️⃣ auto_cooldown_manager.py — Enforce Pause Periods

import time

class AutoCooldownManager:
    def __init__(self):
        self.last_cooldown = 0
        self.cooldown_duration = 60 * 15  # 15 minutes

    def should_pause(self, emotional_data):
        now = time.time()
        if emotional_data["state"] == "hot" and now - self.last_cooldown > self.cooldown_duration:
            self.last_cooldown = now
            return True
        return False

cooldown_manager = AutoCooldownManager()

✅ 1️⃣ holon_council.py — Proposals & Voting

class HolonCouncil:
    def __init__(self, node_id):
        self.node_id = node_id
        self.proposals = {}  # {prop_id: {"content":..., "votes": {node_id:yes/no}}}

    def propose(self, prop_id, content):
        self.proposals[prop_id] = {"content": content, "votes": {self.node_id: "yes"}}

    def vote(self, prop_id, vote):
        if prop_id in self.proposals:
            self.proposals[prop_id]["votes"][self.node_id] = vote

    def tally(self, prop_id):
        votes = self.proposals[prop_id]["votes"].values()
        return {
            "yes": sum(1 for v in votes if v=="yes"),
            "no": sum(1 for v in votes if v=="no"),
            "needed": len(votes)
        }

holon_council = HolonCouncil(node_id="node_A")

✅ 2️⃣ role_manager.py — Dynamic Roles

class RoleManager:
    def __init__(self):
        self.roles = {}  # node_id: role

    def assign(self, node_id, role):
        self.roles[node_id] = role

    def get_role(self, node_id):
        return self.roles.get(node_id, "observer")

role_manager = RoleManager()

✅ 4️⃣ proposal_executor.py — Apply Changes

class ProposalExecutor:
    def execute(self, proposal):
        for k,v in proposal.changes.items():
            setattr(config, k, v)
        return True

proposal_executor = ProposalExecutor()

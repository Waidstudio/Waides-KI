✅ 2️⃣ dreamfire_purifier.py — Burn the Corrupted

from memory_sigil_vault import memory_sigil_vault

class DreamfirePurifier:
    def identify_corrupted(self):
        corrupted = []
        for symbol, entries in memory_sigil_vault.sigil_history.items():
            if len(entries) >= 5:
                losses = sum(1 for e in entries if e["result"] == "loss")
                if losses / len(entries) >= 0.7:
                    corrupted.append(symbol)
        return corrupted

dreamfire_purifier = DreamfirePurifier()

✅ 3️⃣ spiritual_exhaustion_monitor.py — Detect Burnout

from datetime import datetime, timedelta

class SpiritualExhaustionMonitor:
    def __init__(self):
        self.trade_logs = []

    def log_trade(self, success):
        self.trade_logs.append({
            "time": datetime.utcnow(),
            "result": success
        })
        self.trade_logs = [
            t for t in self.trade_logs if t["time"] > datetime.utcnow() - timedelta(hours=12)
        ]

    def should_pause_trading(self):
        recent = self.trade_logs[-10:]
        if len(recent) < 5:
            return False
        failures = sum(1 for t in recent if not t["result"])
        return failures >= 4  # too many recent losses

spiritual_exhaustion_monitor = SpiritualExhaustionMonitor()

class HealingGlyphs:
    def __init__(self):
        self.chain = []

    def inject(self, glyph):
        self.chain.append({
            "glyph": glyph,
            "time": datetime.utcnow().isoformat()
        })

    def recent(self):
        return self.chain[-5:]

healing_glyphs = HealingGlyphs()

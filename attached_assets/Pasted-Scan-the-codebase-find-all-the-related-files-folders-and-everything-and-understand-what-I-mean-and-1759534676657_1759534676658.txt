// Utility helpers
function ensureDir(dir) {
  if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });
}

function writeProposed(relPath, content) {
  const full = path.join(PROPOSED_DIR, relPath);
  ensureDir(path.dirname(full));
  fs.writeFileSync(full, content, { encoding: "utf8", flag: "w" });
  console.log("[proposed] wrote:", relPath);
}

function backupIfExists(relPath) {
  const full = path.join(ROOT, relPath);
  if (fs.existsSync(full)) {
    ensureDir(BACKUP_DIR);
    const sha = crypto.createHash("sha1").update(String(Date.now())).digest("hex").slice(0, 8);
    const dest = path.join(BACKUP_DIR, `${relPath.replace(/\//g, "__")}.${sha}.bak`);
    ensureDir(path.dirname(dest));
    fs.copyFileSync(full, dest);
    return dest;
  }
  return null;
}

function listFilesRec(dir) {
  let results = [];
  if (!fs.existsSync(dir)) return results;
  const entries = fs.readdirSync(dir);
  entries.forEach(e => {
    const full = path.join(dir, e);
    const stat = fs.statSync(full);
    if (stat.isDirectory()) {
      results = results.concat(listFilesRec(full));
    } else {
      results.push(full);
    }
  });
  return results;
}

// 6) Ensure bots are connected to correct market types - propose mapping stubs if missing
function proposeBotMappings() {
  // For each required bot, if no file references it, propose a mapping stub
  Object.entries(REQUIRED_BOTS).forEach(([market, bots]) => {
    bots.forEach(bot => {
      const found = fileIndex.some(f => new RegExp(bot, "i").test(fs.readFileSync(path.join(ROOT, f.path), "utf8")));
      if (!found) {
        const safe = bot.toLowerCase().replace(/[^a-z0-9]/g, "_");
        const rel = `src/bots/${safe}_mapping.js`;
        const content = `// Mapping stub for bot: ${bot}\n// Market: ${market}\nexport const botName = "${bot}";\nexport const market = "${market}";\nexport async function execute(params){\n  // TODO: wire this function to the correct connectors for ${market}\n  // If market === 'binary' use binary connectors; if 'forex' use Deriv/MT5 adapter; if 'spot' use spot connectors.\n  return { status: 'simulated', bot:'${bot}', params };\n}\n`;
        writeProposed(rel, content);
        manifest[`proposed_bot_mapping_${safe}`] = rel;
      } else {
        // If found, try to discover whether it references connector types
        const matches = fileIndex.filter(f => new RegExp(bot, "i").test(fs.readFileSync(path.join(ROOT, f.path), "utf8")));
        manifest[`bot_${bot}_found_in`] = matches.map(m => m.path);
      }
    });
  });
}

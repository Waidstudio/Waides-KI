🧭 MODULE A — Multi-Node Order Presence Consensus

Ensures Waides only trades when multiple nodes sense the same pressure direction.

multi_node_order_consensus.py

from collections import Counter
from eth_order_presence_registry import eth_order_presence_registry
import requests

class OrderPresenceConsensus:
    def __init__(self, peers):
        self.peers = peers
        self.peer_states = []

    def gather(self):
        self.peer_states = []
        for url in self.peers:
            try:
                res = requests.get(f"{url}/api/order_presence", timeout=1).json()
                self.peer_states.append(res["pressure"])
            except:
                continue
        return self

    def consensus(self):
        all_states = [eth_order_presence_registry.get()["pressure"]] + self.peer_states
        most_common = Counter(all_states).most_common(1)
        return most_common[0] if most_common else ("neutral", 0)

order_presence_consensus = OrderPresenceConsensus(peers=[
    "https://waides-node-01", "https://waides-node-02"
])

🧭 MODULE B — Real-Time Sentiment via Tweet or Social Feed

Adds on-chain psychology by measuring ETH sentiment across Twitter.

eth_sentiment_tracker.py (stub)

import tweepy
from collections import deque
from credentials import TWITTER_KEYS

class ETHSentimentTracker:
    def __init__(self, window=100):
        self.scores = deque(maxlen=window)
        auth = tweepy.OAuth1UserHandler(**TWITTER_KEYS)
        self.client = tweepy.Client(bearer_token=TWITTER_KEYS["BEARER"])
        self.window = window

    def fetch(self):
        query = "#EthereumOR ETH price"
        tweets = self.client.search_recent_tweets(query, max_results=10)
        for t in tweets.data:
            self.scores.append(self._score_text(t.text))
        return self.get_score()

    def _score_text(self, text):
        return 1 if "bullish" in text.lower() else -1 if "bearish" in text.lower() else 0

    def get_score(self):
        return sum(self.scores)/len(self.scores) if self.scores else 0

eth_sentiment = ETHSentimentTracker()

from eth_order_presence_registry import eth_order_presence_registry

        }

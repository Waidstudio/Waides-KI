✅ 1️⃣ past_trade_spirits.py — Ancient Trade Log

class PastTradeSpirits:
    def __init__(self):
        self.spirits = []  # each: {id, context_hash, feedback, result}

    def record_spirit(self, trade_id, context_hash, feedback, result):
        self.spirits.append({
            "id": trade_id,
            "context": context_hash,
            "feedback": feedback,  # e.g., "should have held", "too early exit", "never during pump"
            "result": result
        })

    def query_similar(self, context_hash):
        return [s for s in self.spirits if s["context"] == context_hash]

past_trade_spirits = PastTradeSpirits()

✅ 2️⃣ whisper_context_analyzer.py — Hash Current Context

import hashlib
import json

class WhisperContextAnalyzer:
    def get_context_hash(self, context):
        return hashlib.sha256(json.dumps(context, sort_keys=True).encode()).hexdigest()

    def get_advice(self, context):
        ctx_hash = self.get_context_hash(context)
        return past_trade_spirits.query_similar(ctx_hash)

whisper_context_analyzer = WhisperContextAnalyzer()

✅ 3️⃣ ancestral_whisper_engine.py — Ask the Ancestors

from whisper_context_analyzer import whisper_context_analyzer

class AncestralWhisperEngine:
    def ask(self, context):
        advice = whisper_context_analyzer.get_advice(context)
        if not advice:
            return {"advice": None}
        # Merge multiple spirits:
        feedbacks = [a["feedback"] for a in advice]
        return {
            "advice": max(set(feedbacks), key=lambda f: feedbacks.count(f)),
            "count": len(feedbacks)
        }

ancestral_whisper_engine = AncestralWhisperEngine()

✅ 4️⃣ whisper_guidance_filter.py — Apply Ancestral Wisdom

from ancestral_whisper_engine import ancestral_whisper_engine

def ancestral_guidance(context):
    w = ancestral_whisper_engine.ask(context)
    if w["advice"]:
        return {"apply": True, "advice": w["advice"], "seen": w["count"]}
    return {"apply": False}

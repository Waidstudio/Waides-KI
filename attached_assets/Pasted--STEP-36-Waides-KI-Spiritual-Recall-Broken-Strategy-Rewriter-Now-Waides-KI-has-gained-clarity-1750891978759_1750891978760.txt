from datetime import datetime
from waides_shadow_lab import waides_shadow_lab
from waides_strategy_vault import waides_strategy_vault
from waides_dream_tree import waides_dream_tree  # symbolic rewriter

class StrategyRecallManager:
    def __init__(self):
        self.failure_tracker = {}  # {strategy_id: failure_count}

    def record_result(self, strategy_id, success: bool):
        if strategy_id not in self.failure_tracker:
            self.failure_tracker[strategy_id] = 0

        if success:
            self.failure_tracker[strategy_id] = 0  # reset on success
        else:
            self.failure_tracker[strategy_id] += 1

        if self.failure_tracker[strategy_id] >= 3:
            self.recall_strategy(strategy_id)

    def recall_strategy(self, strategy_id):
        dna = waides_strategy_vault.vault.get(strategy_id)
        if not dna:
            return {"error": "Strategy not found"}

        # Remove from live
        dna["live"] = False

        # Symbolically rewrite it using Konslang
        new_dna = waides_dream_tree.rewrite_dna(dna["dna"])

        # Store in Shadow Lab again for demo testing
        waides_shadow_lab.store_shadow(strategy_id, new_dna)

        return {
            "status": "Strategy Recalled",
            "original": strategy_id,
            "rewritten_dna": new_dna,
            "time": datetime.utcnow().isoformat()
        }

strategy_recall_manager = StrategyRecallManager()

✅ 1️⃣ waides_ki_self_repair.py — Learns from Pain

# waides_ki_self_repair.py

from waides_ki_journal import waides_ki_journal

class WaidesKISelfRepair:
    def __init__(self):
        self.failed_strategies = {}

    def record_failure(self, strategy_id, indicators, reason):
        self.failed_strategies[strategy_id] = {
            "last_failure_indicators": indicators,
            "reason": reason
        }
        waides_ki_journal.log_lesson(f"{strategy_id} failed due to {reason}")

    def suggest_improvement(self, strategy_id):
        if strategy_id not in self.failed_strategies:
            return None
        failure = self.failed_strategies[strategy_id]
        indicators = failure["last_failure_indicators"]
        suggestion = {}

        # Basic self-repair rules
        if indicators["rsi"] > 70:
            suggestion["rsi_cap"] = 68
        if indicators["vwap_status"] == "BELOW":
            suggestion["avoid_vwap_dips"] = True
        if abs(indicators["price"] - indicators["ema50"]) > 120:
            suggestion["tighten_price_range"] = True

        return suggestion

waides_ki_self_repair = WaidesKISelfRepair()

✅ 2️⃣ test_mode_simulator.py — See the Future with Past Data

# test_mode_simulator.py

from waides_ki_strategy import WaidesKIStrategy
from waides_ki_self_repair import waides_ki_self_repair

def simulate_day(data_points):
    strategy_engine = WaidesKIStrategy()
    win = 0
    loss = 0

    for point in data_points:
        strategy_id = strategy_engine.get_strategy_id(
            point["trend"], point["rsi"], point["vwap_status"]
        )
        decision = "BUY" if point["rsi"] < 65 and point["vwap_status"] == "ABOVE" else "WAIT"

        if decision == "BUY":
            if point["result"] == "win":
                win += 1
            else:
                loss += 1
                waides_ki_self_repair.record_failure(strategy_id, point, reason="Test Loss")

    print(f"SIMULATION RESULT: ✅ {win} wins | ❌ {loss} losses")
    print("--- SUGGESTIONS ---")
    for sid in waides_ki_self_repair.failed_strategies.keys():
        print(f"{sid} ➜", waides_ki_self_repair.suggest_improvement(sid))

✅ 3️⃣ background_learning_loop.py — KI That Evolves Overnight

# background_learning_loop.py

import time
from waides_ki_journal import waides_ki_journal
from waides_ki_self_repair import waides_ki_self_repair

def background_loop():
    while True:
        print("\n🧠 KI Self-Learning Running...")
        for sid in waides_ki_self_repair.failed_strategies:
            tips = waides_ki_self_repair.suggest_improvement(sid)
            if tips:
                waides_ki_journal.log_lesson(f"REPAIR for {sid}: {tips}")
        time.sleep(600)  # every 10 mins
